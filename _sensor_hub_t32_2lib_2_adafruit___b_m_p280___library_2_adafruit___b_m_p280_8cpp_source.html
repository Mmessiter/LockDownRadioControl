336 float Adafruit_BMP280::readPressure() {
337  int64_t var1, var2, p;
338 
339  // Must be done first to get the t_fine variable set up
340  readTemperature();
341 
342  int32_t adc_P = read24(BMP280_REGISTER_PRESSUREDATA);
343  adc_P >>= 4;
344 
345  var1 = ((int64_t)t_fine) - 128000;
346  var2 = var1 * var1 * (int64_t)_bmp280_calib.dig_P6;
347  var2 = var2 + ((var1 * (int64_t)_bmp280_calib.dig_P5) << 17);
348  var2 = var2 + (((int64_t)_bmp280_calib.dig_P4) << 35);
349  var1 = ((var1 * var1 * (int64_t)_bmp280_calib.dig_P3) >> 8) +
350  ((var1 * (int64_t)_bmp280_calib.dig_P2) << 12);
351  var1 =
352  (((((int64_t)1) << 47) + var1)) * ((int64_t)_bmp280_calib.dig_P1) >> 33;
353 
354  if (var1 == 0) {
355  return 0; // avoid exception caused by division by zero
356  }
357  p = 1048576 - adc_P;
358  p = (((p << 31) - var2) * 3125) / var1;
359  var1 = (((int64_t)_bmp280_calib.dig_P9) * (p >> 13) * (p >> 13)) >> 25;
360  var2 = (((int64_t)_bmp280_calib.dig_P8) * p) >> 19;
361 
362  p = ((p + var1 + var2) >> 8) + (((int64_t)_bmp280_calib.dig_P7) << 4);
363  return (float)p / 256;
364 }
