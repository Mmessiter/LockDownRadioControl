9FASTRUN bool CheckTXVolts()
10{
11 char JTX[] = "JTX"; // Labels on Nextion
12 char FrontView_TXBV[] = "TXBV"; // Labels on Nextion
13 bool TXWarningFlag = false;
14 float TransmitterBatteryPercentLeft, TransmitterBatteryVolts;
15 char TXBattInfo[80];
16 char pc[] = "%";
17 char nbuf[10]; // Little buffer for numbers
18 char v[] = "V";
19 char t17[] = "t17";
20
21 if (USE_INA219) {
22 TransmitterBatteryVolts = ((ina219.getBusVoltage_V()) * 100) + (TxVoltageCorrection * 2); // Correction for inaccurate ina219
23 dtostrf(TransmitterBatteryVolts / 200, 2, 2, nbuf); // Volts per cell
24 if (TXLiPo) { // Does TX have a LiPo or a LiFePo4?
25 TransmitterBatteryPercentLeft = map(TransmitterBatteryVolts, 3.5 * 200, 4.00 * 200, 0, 100); // LIPO Battery 3.50 -> c. 4.00 volts per cell
26 }
27 else { // No, it's a LiFePo4
28 TransmitterBatteryPercentLeft = map(TransmitterBatteryVolts, 3.2 * 200, 3.33 * 200, 0, 100); // LiFePo4 Battery 3.1 -> 3.35 volts per cell
29 }
30 if (TransmitterBatteryPercentLeft < LowBattery) {
31 TXWarningFlag = true;
32 WarningSound = BATTERYISLOW;
33 }
34 TransmitterBatteryPercentLeft = constrain(TransmitterBatteryPercentLeft, 0, 100);
35 strcat(TXBattInfo, pc);
36 if (CurrentView == FRONTVIEW) {
37 SendValue(JTX, TransmitterBatteryPercentLeft);
38 strcat(nbuf, v);
39 SendText(FrontView_TXBV, nbuf);
40 }
41 if (CurrentView == DATAVIEW) {
42 SendText(t17, nbuf);
43 }
44 }
45
46 return TXWarningFlag;
47}

51FASTRUN bool CheckRXVolts()
52{
53 static bool RXWarningFlag = false;
54 float ReadVolts = 0;
55 uint8_t GreenPercentBar = 0;
56 char JRX[] = "JRX";
57
58 char Vbuf[10];
59 char RXBattInfo[80];
60 float VoltsPerCell = 0;
61 char FrontView_RXBV[] = "RXBV";
62 char RXPC[] = "RXPC";
63 char PerCell[] = " per cell)";
64 char RXBattNA[] = ""; //"(No data from RX)";
65 char v[] = "V (";
66 char pc[] = "%";
67 char spaces[] = " ";
68 char t6[] = "t6";
69
70 ReadVolts = (RXModelVolts * 100) + (RxVoltageCorrection * RXCellCount);
71 GreenPercentBar = map(ReadVolts, 3.4f * RXCellCount * 100, 4.2f * RXCellCount * 100, 0, 100);
72 if (RXVoltsDetected) {
73 GreenPercentBar = constrain(GreenPercentBar, 0, 100);
74 WarningSound = BATTERYISLOW;
75 if (BoundFlag) {
76 VoltsPerCell = (ReadVolts / RXCellCount) / 100;
77 if (CurrentView == FRONTVIEW) {
78 SendValue(JRX, GreenPercentBar);
79 strcat(Str(Vbuf, GreenPercentBar, 0), pc);
80 SendText(RXPC, Vbuf);
81 strcpy(RXBattInfo, ModelVolts);
82 strcat(RXBattInfo, v);
83 dtostrf(VoltsPerCell, 2, 2, Vbuf);
84 strcat(RXBattInfo, Vbuf);
85 strcat(RXBattInfo, PerCell);
86 SendText(FrontView_RXBV, RXBattInfo);
87 }
88 if (CurrentView == DATAVIEW) {
89 dtostrf(VoltsPerCell, 2, 2, Vbuf);
90 SendText(t6, Vbuf);
91 }
92
93 if ((VoltsPerCell <= StopFlyingVoltsPerCell) && (BoundFlag && ModelMatched)) {
94 RXWarningFlag = true;
95 WarningSound = STORAGECHARGE;
96 }
97 if (VoltsPerCell > StopFlyingVoltsPerCell) {
98 RXWarningFlag = false; // Reset warning as voltage recovered
99 }
100 }
101 }
102 else {
103 if (BoundFlag && CurrentView == FRONTVIEW) {
104 SendText(FrontView_RXBV, RXBattNA);
105 SendValue(JRX, 0);
106 SendText(RXPC, spaces);
107 }
108 }
109 return RXWarningFlag;
110}
