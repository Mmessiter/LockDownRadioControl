14/*
15 * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
16 * @param uncompressed_buf[in]
17 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
18 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
19 */
20void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, uint8_t uncompressed_size)
21{
22 uint8_t p = 0;
23 for (uint8_t l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
24 uncompressed_buf[p] = compressed_buf[l] >> 4;
25 ++p;
26 uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
27 ++p;
28 uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
29 ++p;
30 uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
31 ++p;
32 }
33}

40/* Compresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
41 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
42 * @param uncompressed_buf[in]
43 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short). This *must* be divisible by 4.
44 */
45FASTRUN void Compress(uint16_t* compressed_buf, uint16_t* uncompressed_buf, uint8_t uncompressed_size)
46{
47 if (NewCompressNeeded) { // no need to recompress old data
48 NewCompressNeeded = false;
49 uint8_t p = 0;
50 for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
51 compressed_buf[l] = uncompressed_buf[p] << 4 | uncompressed_buf[p + 1] >> 8;
52 ++p;
53 compressed_buf[l + 1] = uncompressed_buf[p] << 8 | uncompressed_buf[p + 1] >> 4;
54 ++p;
55 compressed_buf[l + 2] = uncompressed_buf[p] << 12 | uncompressed_buf[p + 1];
56 ++p;
57 ++p;
58 }
59 }

64void LoadParameters()
65{
66 uint16_t Twobytes = 0;
67 uint8_t FS_Byte1;
68 uint8_t FS_Byte2;
69
70 switch (Parameters.ID) {
71
72 case 1:
73 Twobytes = MakeTwobytes(FailSafeChannel); // 16 bool values compressed to 16 bits
74 FS_Byte1 = uint8_t(Twobytes >> 8); // Send as two bytes
75 FS_Byte2 = uint8_t(Twobytes & 0x00FF);
76 Parameters.word1 = FS_Byte1; // These are failsafe flags
77 Parameters.word2 = FS_Byte2; // These are failsafe flags
78 break;
79
80 case 2:
81 Parameters.word1 = Qnh >> 8; // (HiByte) Qnh is current atmospheric pressure at sea level here (an aviation term)
82 Parameters.word2 = Qnh & 0x00ff; // (LowByte) Qnh is current atmospheric pressure at sea level here (an aviation term)
83 break;
84
85 case 3:
86 if (GPSMarkHere) {
87 Parameters.word1 = 0;
88 Parameters.word2 = GPSMarkHere;
89 GPSMarkHere = 0;
90 }
91 break;
92 case 4:
93 Parameters.word1 = ModelMatched; // let receiver know whether correct model is loaded.
94 Parameters.word2 = SwapWaveBand;
95 if (SwapWaveBand == 2) SetTestFrequencies();
96 if (SwapWaveBand == 1) SetUKFrequencies();
97 SwapWaveBand = 0;
98 break;
99 case 5:
100 Parameters.word1 = PPMdata.UseSBUSFromRX; // 1 - 0
101 Parameters.word2 = PPMdata.PPMChannelCount;
102 break;
103 default:
104 break;
105 }
106}

223uint8_t EncodeTheChangedChannels(){
224 uint8_t NumberOfChangedChannels = 0;
225 DataTosend.ChannelBitMask = 0; // clear the ChannelBitMask 16 BIT WORD
226 for (int i = 0; i < CHANNELSUSED; ++i){ // check for changed channels
227 if ((SendBuffer[i] != PreviousBuffer[i]) && (NumberOfChangedChannels < 4)) { // !!! 4 is the maximum number of channels that can be sent in one packet !!!! <<< *****
228 RawDataBuffer[NumberOfChangedChannels] = SendBuffer[i]; // load a changed channel into the rawdatabuffer
229 PrePreviousBuffer[i] = PreviousBuffer[i]; // save previous buffer incase we need to repeat it
230 PreviousBuffer[i] = SendBuffer[i]; // save it for next time in case it succeeds this time
231 DataTosend.ChannelBitMask |= (1 << i); // set the bit in the ChannelBitMask word
232 ++NumberOfChangedChannels; // increment the rawdatabuffer index
233 }
234 }
235 return NumberOfChangedChannels; // Return the number of channels that have changed
236}

241FASTRUN void SendData()
242{
243 if (SendNoData) return;
244 if ((millis() - LastPacketSentTime) >= FHSS_data::PaceMaker) {
245 LastPacketSentTime = millis();
246 if (BuddyPupilOnPPM) {SendViaPPM(); return;} // If buddying (SLAVE) by wire, send SBUS data down wire only and transmit nothing.
247 Connected = false; // Assume failure until an ACK is received.
248 FlushFifos(); // This flush avoids a lockup that happens when the FIFO gets full.
249 uint8_t NumberOfChangedChannels = EncodeTheChangedChannels(); // Returns the number of channels that have changed, as well as loading the raw data buffer with the changed channels
250 if (AddExtraParameters) NumberOfChangedChannels = SendExtraParamemters(NumberOfChangedChannels); // Add parameters here if there are some to go ...
251 uint8_t ByteCountToTransmit = ((float) NumberOfChangedChannels * 1.5f) + 4; // 1.5 is the compression ratio. 2 is the number of extra bytes for flags - plus 1 word because int rounds downwards.
252 uint8_t SizeOfUnCompressedData = (ByteCountToTransmit / 1.5) ;
253 Compress(DataTosend.CompressedData, RawDataBuffer, SizeOfUnCompressedData); // Compress the raw data buffer into the compressed data buffer (reduces it to 75% of original size)
254 if (Radio1.write(&DataTosend, ByteCountToTransmit)) {SuccessfulPacket();} else {FailedPacket();} // Send the data packet complete with ChannelBitMask and compressed data
255 }else{
256 if (BuddyMasterOnWireless) SendSpecialPacket(); // takes about 4 - 5 ms. Gets buddy control data in ACK payload
257 }
258 if ((!Connected) && (BuddyMasterOnWireless)) SendSpecialPacket(); // So Pupil knows we are still here
259}

484uint8_t CheckPipeNibbles(uint8_t b)
485{
486 uint8_t temp;
487 uint8_t BadLowerNibble[BADNIBBLECOUNT] = {0x05, 0x0a, 0x02, 0x01, 0x00, 0x0f};
488 uint8_t BadHigherNibble[BADNIBBLECOUNT] = {0x50, 0xa0, 0x20, 0x10, 0x00, 0xf0};
489 uint8_t BetterLowerNibble[BADNIBBLECOUNT] = {0x03, 0x04, 0x06, 0x07, 0x08, 0x09};
490 uint8_t BetterHigherNibble[BADNIBBLECOUNT] = {0x30, 0x40, 0x60, 0x70, 0x80, 0x90};
491 if (!b) return 0x36; // return an acceptable byte for a zero
492
493 for (int i = 0; i < BADNIBBLECOUNT; ++i) { // ********** check LOWER nibble **********
494 if ((b & 0x0f) == BadLowerNibble[i])
495 {
496 temp = b & 0xf0; // save only the hi nibble in temp
497 b = temp | BetterLowerNibble[i]; // put an acceptable nibble into lower nibble
498 }
499 }
500 for (int i = 0; i < BADNIBBLECOUNT; ++i) { // ********** check HIGHER nibble **********
501 if ((b & 0xf0) == BadHigherNibble[i])
502 {
503 temp = b & 0x0f; // save only the Low nibble in temp
504 b = temp | BetterHigherNibble[i]; // put an acceptable nibble into Higher nibble
505 }
506 }
507 return b;
508}
