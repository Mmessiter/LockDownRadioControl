13void SaveCheckSum32()
14{ // uses 5 bytes. Last one is indicator of use.
15
16 DoingCheckSm = true;
17 SDUpdate32BITS(SDCardAddress, FileCheckSum);
18 SDCardAddress += 4;
19 SDUpdate8BITS(SDCardAddress, 0xFF); // indicator
20 SDCardAddress++;
21 DoingCheckSm = false;
22
23#ifdef DB_CHECKSUM
24 Serial.print("Writing: ");
25 Serial.println(FileCheckSum);
26#endif
27}

29/*********************************************************************************************************************************/
30void ReadCheckSum32()
31{ // uses 5 bytes. Last one is indicator of use.
32 // This function sets ErrorState to a non zero value if there'a a file error
33
34 bool UseCheckSm = false;
35 DoingCheckSm = true;
36 uint32_t ch;
37 ch = SDRead32BITS(SDCardAddress);
38 SDCardAddress += 4;
39 if (SDRead8BITS(SDCardAddress) == 0xFF) UseCheckSm = true; // if this byte isn't 0xFF then checksum was never written here.
40 ++SDCardAddress;
41 DoingCheckSm = false;
42 if (UseCheckSm) {
43 if (ch != FileCheckSum) {
44 ErrorState = CHECKSUMERROR;
45 }
46#ifdef DB_CHECKSUM
47 Serial.print("Read from file: ");
48 Serial.println(ch);
49 if (ch == FileCheckSum) {
50 Serial.println("FILE CHECKSUM IS GOOD :-) ! ");
51 }
52 else {
53 Serial.print("FILE CHECKSUM WAS ");
54 Serial.println(FileCheckSum);
55 }
56#endif
57 }
58}

71void CheckOutPutChannels() // This function checks for bad or duplicate output channels and resets them if found
72{
73 bool resetit = false;
74 for (int i = 0; i < 16; ++i) {
75 if ((ChannelOutPut[i] > 15) || (!CheckDuplicate(ChannelOutPut[i], i)))
76 {
77 resetit = true;
78 break;
79 }
80 }
81 if (resetit) {
82 for (int i = 0; i < 16; ++i) {
83 ChannelOutPut[i] = i;
84 }
85 }

465void BuildCheckSum(int p_address, short int p_value)
466{
467 if (!DoingCheckSm) FileCheckSum += (p_value * (p_address + 1)); // don't include checksum in its own calculation
468}

470/*********************************************************************************************************************************/
471
472void SDUpdate32BITS(int p_address, uint32_t p_value)
473{
474 BuildCheckSum(p_address, p_value);
475 ModelsFileNumber.seek(p_address);
476 ShortDelay();
477 ModelsFileNumber.write(uint8_t(p_value));
478 ShortDelay();
479 ModelsFileNumber.write(uint8_t(p_value >> 8));
480 ShortDelay();
481 ModelsFileNumber.write(uint8_t(p_value >> 16));
482 ShortDelay();
483 ModelsFileNumber.write(uint8_t(p_value >> 24));
484 ShortDelay();
485}

487/*********************************************************************************************************************************/
488
489uint32_t SDRead32BITS(int p_address)
490{
491 ModelsFileNumber.seek(p_address);
492 uint32_t r = ModelsFileNumber.read();
493 r += ModelsFileNumber.read() << 8;
494 r += ModelsFileNumber.read() << 16;
495 r += ModelsFileNumber.read() << 24;
496 return r;
497}
