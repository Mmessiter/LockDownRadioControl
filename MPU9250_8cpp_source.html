37/* starts communication with the MPU-9250 */
38int MPU9250::begin(){
39 if( _useSPI ) { // using SPI for communication
40 // use low speed SPI for register setting
41 _useSPIHS = false;
42 // setting CS pin to output
43 pinMode(_csPin,OUTPUT);
44 // setting CS pin high
45 digitalWrite(_csPin,HIGH);
46 // begin SPI communication
47 _spi->begin();
48 } else { // using I2C for communication
49 // starting the I2C bus
50 _i2c->begin();
51 // setting the I2C clock
52 _i2c->setClock(_i2cRate);
53 }
54 // select clock source to gyro
55 if(writeRegister(PWR_MGMNT_1,CLOCK_SEL_PLL) < 0){
56 return -1;
57 }
58 // enable I2C master mode
59 if(writeRegister(USER_CTRL,I2C_MST_EN) < 0){
60 return -2;
61 }
62 // set the I2C bus speed to 400 kHz
63 if(writeRegister(I2C_MST_CTRL,I2C_MST_CLK) < 0){
64 return -3;
65 }
66 // set AK8963 to Power Down
67 writeAK8963Register(AK8963_CNTL1,AK8963_PWR_DOWN);
68 // reset the MPU9250
69 writeRegister(PWR_MGMNT_1,PWR_RESET);
70 // wait for MPU-9250 to come back up
71 delay(1);
72 // reset the AK8963
73 writeAK8963Register(AK8963_CNTL2,AK8963_RESET);
74 // select clock source to gyro
75 if(writeRegister(PWR_MGMNT_1,CLOCK_SEL_PLL) < 0){
76 return -4;
77 }
78 // check the WHO AM I byte, expected value is 0x71 (decimal 113) or 0x73 (decimal 115)
79 if((whoAmI() != 113)&&(whoAmI() != 115)){
80 return -5;
81 }
82 // enable accelerometer and gyro
83 if(writeRegister(PWR_MGMNT_2,SEN_ENABLE) < 0){
84 return -6;
85 }
86 // setting accel range to 16G as default
87 if(writeRegister(ACCEL_CONFIG,ACCEL_FS_SEL_16G) < 0){
88 return -7;
89 }
90 _accelScale = G * 16.0f/32767.5f; // setting the accel scale to 16G
91 _accelRange = ACCEL_RANGE_16G;
92 // setting the gyro range to 2000DPS as default
93 if(writeRegister(GYRO_CONFIG,GYRO_FS_SEL_2000DPS) < 0){
94 return -8;
95 }
96 _gyroScale = 2000.0f/32767.5f * _d2r; // setting the gyro scale to 2000DPS
97 _gyroRange = GYRO_RANGE_2000DPS;
98 // setting bandwidth to 184Hz as default
99 if(writeRegister(ACCEL_CONFIG2,ACCEL_DLPF_184) < 0){
100 return -9;
101 }
102 if(writeRegister(CONFIG,GYRO_DLPF_184) < 0){ // setting gyro bandwidth to 184Hz
103 return -10;
104 }
105 _bandwidth = DLPF_BANDWIDTH_184HZ;
106 // setting the sample rate divider to 0 as default
107 if(writeRegister(SMPDIV,0x00) < 0){
108 return -11;
109 }
110 _srd = 0;
111 // enable I2C master mode
112 if(writeRegister(USER_CTRL,I2C_MST_EN) < 0){
113 return -12;
114 }
115 // set the I2C bus speed to 400 kHz
116 if( writeRegister(I2C_MST_CTRL,I2C_MST_CLK) < 0){
117 return -13;
118 }
119 // check AK8963 WHO AM I register, expected value is 0x48 (decimal 72)
120 if( whoAmIAK8963() != 72 ){
121 return -14;
122 }
123 /* get the magnetometer calibration */
124 // set AK8963 to Power Down
125 if(writeAK8963Register(AK8963_CNTL1,AK8963_PWR_DOWN) < 0){
126 return -15;
127 }
128 delay(100); // long wait between AK8963 mode changes
129 // set AK8963 to FUSE ROM access
130 if(writeAK8963Register(AK8963_CNTL1,AK8963_FUSE_ROM) < 0){
131 return -16;
132 }
133 delay(100); // long wait between AK8963 mode changes
134 // read the AK8963 ASA registers and compute magnetometer scale factors
135 readAK8963Registers(AK8963_ASA,3,_buffer);
136 _magScaleX = ((((float)_buffer[0]) - 128.0f)/(256.0f) + 1.0f) * 4912.0f / 32760.0f; // micro Tesla
137 _magScaleY = ((((float)_buffer[1]) - 128.0f)/(256.0f) + 1.0f) * 4912.0f / 32760.0f; // micro Tesla
138 _magScaleZ = ((((float)_buffer[2]) - 128.0f)/(256.0f) + 1.0f) * 4912.0f / 32760.0f; // micro Tesla
139 // set AK8963 to Power Down
140 if(writeAK8963Register(AK8963_CNTL1,AK8963_PWR_DOWN) < 0){
141 return -17;
142 }
143 delay(100); // long wait between AK8963 mode changes
144 // set AK8963 to 16 bit resolution, 100 Hz update rate
145 if(writeAK8963Register(AK8963_CNTL1,AK8963_CNT_MEAS2) < 0){
146 return -18;
147 }
148 delay(100); // long wait between AK8963 mode changes
149 // select clock source to gyro
150 if(writeRegister(PWR_MGMNT_1,CLOCK_SEL_PLL) < 0){
151 return -19;
152 }
153 // instruct the MPU9250 to get 7 bytes of data from the AK8963 at the sample rate
154 readAK8963Registers(AK8963_HXL,7,_buffer);
155 // estimate gyro bias
156 if (calibrateGyro() < 0) {
157 return -20;
158 }
159 // successful init, return 1
160 return 1;
161}

312int MPU9250::setSrd(uint8_t srd) {
313 // use low speed SPI for register setting
314 _useSPIHS = false;
315 /* setting the sample rate divider to 19 to facilitate setting up magnetometer */
316 if(writeRegister(SMPDIV,19) < 0){ // setting the sample rate divider
317 return -1;
318 }
319 if(srd > 9){
320 // set AK8963 to Power Down
321 if(writeAK8963Register(AK8963_CNTL1,AK8963_PWR_DOWN) < 0){
322 return -2;
323 }
324 delay(100); // long wait between AK8963 mode changes
325 // set AK8963 to 16 bit resolution, 8 Hz update rate
326 if(writeAK8963Register(AK8963_CNTL1,AK8963_CNT_MEAS1) < 0){
327 return -3;
328 }
329 delay(100); // long wait between AK8963 mode changes
330 // instruct the MPU9250 to get 7 bytes of data from the AK8963 at the sample rate
331 readAK8963Registers(AK8963_HXL,7,_buffer);
332 } else {
333 // set AK8963 to Power Down
334 if(writeAK8963Register(AK8963_CNTL1,AK8963_PWR_DOWN) < 0){
335 return -2;
336 }
337 delay(100); // long wait between AK8963 mode changes
338 // set AK8963 to 16 bit resolution, 100 Hz update rate
339 if(writeAK8963Register(AK8963_CNTL1,AK8963_CNT_MEAS2) < 0){
340 return -3;
341 }
342 delay(100); // long wait between AK8963 mode changes
343 // instruct the MPU9250 to get 7 bytes of data from the AK8963 at the sample rate
344 readAK8963Registers(AK8963_HXL,7,_buffer);
345 }
346 /* setting the sample rate divider */

348 return -4;

378/* configures and enables wake on motion, low power mode */
379int MPU9250::enableWakeOnMotion(float womThresh_mg,LpAccelOdr odr) {
380 // use low speed SPI for register setting
381 _useSPIHS = false;
382 // set AK8963 to Power Down
383 writeAK8963Register(AK8963_CNTL1,AK8963_PWR_DOWN);
384 // reset the MPU9250
385 writeRegister(PWR_MGMNT_1,PWR_RESET);
386 // wait for MPU-9250 to come back up
387 delay(1);
388 if(writeRegister(PWR_MGMNT_1,0x00) < 0){ // cycle 0, sleep 0, standby 0
389 return -1;
390 }
391 if(writeRegister(PWR_MGMNT_2,DIS_GYRO) < 0){ // disable gyro measurements
392 return -2;
393 }
394 if(writeRegister(ACCEL_CONFIG2,ACCEL_DLPF_184) < 0){ // setting accel bandwidth to 184Hz
395 return -3;
396 }
397 if(writeRegister(INT_ENABLE,INT_WOM_EN) < 0){ // enabling interrupt to wake on motion
398 return -4;
399 }
400 if(writeRegister(MOT_DETECT_CTRL,(ACCEL_INTEL_EN | ACCEL_INTEL_MODE)) < 0){ // enabling accel hardware intelligence
401 return -5;
402 }
403 _womThreshold = map(womThresh_mg, 0, 1020, 0, 255);
404 if(writeRegister(WOM_THR,_womThreshold) < 0){ // setting wake on motion threshold
405 return -6;
406 }
407 if(writeRegister(LP_ACCEL_ODR,(uint8_t)odr) < 0){ // set frequency of wakeup
408 return -7;
409 }
410 if(writeRegister(PWR_MGMNT_1,PWR_CYCLE) < 0){ // switch to accel low power mode
411 return -8;
412 }
413 return 1;
414}

416/* configures and enables the FIFO buffer */
417int MPU9250FIFO::enableFifo(bool accel,bool gyro,bool mag,bool temp) {
418 // use low speed SPI for register setting
419 _useSPIHS = false;
420 if(writeRegister(USER_CTRL, (0x40 | I2C_MST_EN)) < 0){
421 return -1;
422 }
423 if(writeRegister(FIFO_EN,(accel*FIFO_ACCEL)|(gyro*FIFO_GYRO)|(mag*FIFO_MAG)|(temp*FIFO_TEMP)) < 0){
424 return -2;
425 }
426 _enFifoAccel = accel;
427 _enFifoGyro = gyro;
428 _enFifoMag = mag;
429 _enFifoTemp = temp;
430 _fifoFrameSize = accel*6 + gyro*6 + mag*7 + temp*2;
431 return 1;
432}

434/* reads the most current data from MPU9250 and stores in buffer */
435int MPU9250::readSensor() {
436 _useSPIHS = true; // use the high speed SPI for data readout
437 // grab the data from the MPU9250
438 if (readRegisters(ACCEL_OUT, 21, _buffer) < 0) {
439 return -1;
440 }
441 // combine into 16 bit values
442 _axcounts = (((int16_t)_buffer[0]) << 8) | _buffer[1];
443 _aycounts = (((int16_t)_buffer[2]) << 8) | _buffer[3];
444 _azcounts = (((int16_t)_buffer[4]) << 8) | _buffer[5];
445 _tcounts = (((int16_t)_buffer[6]) << 8) | _buffer[7];
446 _gxcounts = (((int16_t)_buffer[8]) << 8) | _buffer[9];
447 _gycounts = (((int16_t)_buffer[10]) << 8) | _buffer[11];
448 _gzcounts = (((int16_t)_buffer[12]) << 8) | _buffer[13];
449 _hxcounts = (((int16_t)_buffer[15]) << 8) | _buffer[14];
450 _hycounts = (((int16_t)_buffer[17]) << 8) | _buffer[16];
451 _hzcounts = (((int16_t)_buffer[19]) << 8) | _buffer[18];
452 // transform and convert to float values
453 _ax = (((float)(tX[0]*_axcounts + tX[1]*_aycounts + tX[2]*_azcounts) * _accelScale) - _axb)*_axs;
454 _ay = (((float)(tY[0]*_axcounts + tY[1]*_aycounts + tY[2]*_azcounts) * _accelScale) - _ayb)*_ays;
455 _az = (((float)(tZ[0]*_axcounts + tZ[1]*_aycounts + tZ[2]*_azcounts) * _accelScale) - _azb)*_azs;
456 _gx = ((float)(tX[0]*_gxcounts + tX[1]*_gycounts + tX[2]*_gzcounts) * _gyroScale) - _gxb;
457 _gy = ((float)(tY[0]*_gxcounts + tY[1]*_gycounts + tY[2]*_gzcounts) * _gyroScale) - _gyb;
458 _gz = ((float)(tZ[0]*_gxcounts + tZ[1]*_gycounts + tZ[2]*_gzcounts) * _gyroScale) - _gzb;
459 _hx = (((float)(_hxcounts) * _magScaleX) - _hxb)*_hxs;
460 _hy = (((float)(_hycounts) * _magScaleY) - _hyb)*_hys;
461 _hz = (((float)(_hzcounts) * _magScaleZ) - _hzb)*_hzs;
462 _t = ((((float) _tcounts) - _tempOffset)/_tempScale) + _tempOffset;
463 return 1;
464}

516/* reads data from the MPU9250 FIFO and stores in buffer */
517int MPU9250FIFO::readFifo() {
518 _useSPIHS = true; // use the high speed SPI for data readout
519 // get the fifo size
520 readRegisters(FIFO_COUNT, 2, _buffer);
521 _fifoSize = (((uint16_t) (_buffer[0]&0x0F)) <<8) + (((uint16_t) _buffer[1]));
522 // read and parse the buffer
523 for (size_t i=0; i < _fifoSize/_fifoFrameSize; i++) {
524 // grab the data from the MPU9250
525 if (readRegisters(FIFO_READ,_fifoFrameSize,_buffer) < 0) {
526 return -1;
527 }
528 if (_enFifoAccel) {
529 // combine into 16 bit values
530 _axcounts = (((int16_t)_buffer[0]) << 8) | _buffer[1];
531 _aycounts = (((int16_t)_buffer[2]) << 8) | _buffer[3];
532 _azcounts = (((int16_t)_buffer[4]) << 8) | _buffer[5];
533 // transform and convert to float values
534 _axFifo[i] = (((float)(tX[0]*_axcounts + tX[1]*_aycounts + tX[2]*_azcounts) * _accelScale)-_axb)*_axs;
535 _ayFifo[i] = (((float)(tY[0]*_axcounts + tY[1]*_aycounts + tY[2]*_azcounts) * _accelScale)-_ayb)*_ays;
536 _azFifo[i] = (((float)(tZ[0]*_axcounts + tZ[1]*_aycounts + tZ[2]*_azcounts) * _accelScale)-_azb)*_azs;
537 _aSize = _fifoSize/_fifoFrameSize;
538 }
539 if (_enFifoTemp) {
540 // combine into 16 bit values
541 _tcounts = (((int16_t)_buffer[0 + _enFifoAccel*6]) << 8) | _buffer[1 + _enFifoAccel*6];
542 // transform and convert to float values
543 _tFifo[i] = ((((float) _tcounts) - _tempOffset)/_tempScale) + _tempOffset;
544 _tSize = _fifoSize/_fifoFrameSize;
545 }
546 if (_enFifoGyro) {
547 // combine into 16 bit values
548 _gxcounts = (((int16_t)_buffer[0 + _enFifoAccel*6 + _enFifoTemp*2]) << 8) | _buffer[1 + _enFifoAccel*6 + _enFifoTemp*2];
549 _gycounts = (((int16_t)_buffer[2 + _enFifoAccel*6 + _enFifoTemp*2]) << 8) | _buffer[3 + _enFifoAccel*6 + _enFifoTemp*2];
550 _gzcounts = (((int16_t)_buffer[4 + _enFifoAccel*6 + _enFifoTemp*2]) << 8) | _buffer[5 + _enFifoAccel*6 + _enFifoTemp*2];
551 // transform and convert to float values
552 _gxFifo[i] = ((float)(tX[0]*_gxcounts + tX[1]*_gycounts + tX[2]*_gzcounts) * _gyroScale) - _gxb;
553 _gyFifo[i] = ((float)(tY[0]*_gxcounts + tY[1]*_gycounts + tY[2]*_gzcounts) * _gyroScale) - _gyb;
554 _gzFifo[i] = ((float)(tZ[0]*_gxcounts + tZ[1]*_gycounts + tZ[2]*_gzcounts) * _gyroScale) - _gzb;
555 _gSize = _fifoSize/_fifoFrameSize;
556 }
557 if (_enFifoMag) {
558 // combine into 16 bit values
559 _hxcounts = (((int16_t)_buffer[1 + _enFifoAccel*6 + _enFifoTemp*2 + _enFifoGyro*6]) << 8) | _buffer[0 + _enFifoAccel*6 + _enFifoTemp*2 + _enFifoGyro*6];

703/* finds bias and scale factor calibration for the accelerometer,
704this should be run for each axis in each direction (6 total) to find
705the min and max values along each */
706int MPU9250::calibrateAccel() {
707 // set the range, bandwidth, and srd
708 if (setAccelRange(ACCEL_RANGE_2G) < 0) {
709 return -1;
710 }
711 if (setDlpfBandwidth(DLPF_BANDWIDTH_20HZ) < 0) {
712 return -2;
713 }
714 if (setSrd(19) < 0) {
715 return -3;
716 }
717
718 // take samples and find min / max
719 _axbD = 0;
720 _aybD = 0;
721 _azbD = 0;
722 for (size_t i=0; i < _numSamples; i++) {
723 readSensor();
724 _axbD += (getAccelX_mss()/_axs + _axb)/((double)_numSamples);
725 _aybD += (getAccelY_mss()/_ays + _ayb)/((double)_numSamples);
726 _azbD += (getAccelZ_mss()/_azs + _azb)/((double)_numSamples);
727 delay(20);
728 }
729 if (_axbD > 9.0f) {
730 _axmax = (float)_axbD;
731 }
732 if (_aybD > 9.0f) {
733 _aymax = (float)_aybD;
734 }
735 if (_azbD > 9.0f) {
736 _azmax = (float)_azbD;
737 }
738 if (_axbD < -9.0f) {
739 _axmin = (float)_axbD;
740 }
741 if (_aybD < -9.0f) {
742 _aymin = (float)_aybD;
743 }
744 if (_azbD < -9.0f) {
745 _azmin = (float)_azbD;
746 }
747
748 // find bias and scale factor
749 if ((abs(_axmin) > 9.0f) && (abs(_axmax) > 9.0f)) {
750 _axb = (_axmin + _axmax) / 2.0f;
751 _axs = G/((abs(_axmin) + abs(_axmax)) / 2.0f);
752 }
753 if ((abs(_aymin) > 9.0f) && (abs(_aymax) > 9.0f)) {
754 _ayb = (_aymin + _aymax) / 2.0f;
755 _ays = G/((abs(_aymin) + abs(_aymax)) / 2.0f);
756 }
757 if ((abs(_azmin) > 9.0f) && (abs(_azmax) > 9.0f)) {
758 _azb = (_azmin + _azmax) / 2.0f;
759 _azs = G/((abs(_azmin) + abs(_azmax)) / 2.0f);
760 }
761
762 // set the range, bandwidth, and srd back to what they were
763 if (setAccelRange(_accelRange) < 0) {
764 return -4;
765 }
766 if (setDlpfBandwidth(_bandwidth) < 0) {
767 return -5;
768 }
769 if (setSrd(_srd) < 0) {
770 return -6;
771 }
772 return 1;
773}

823/* finds bias and scale factor calibration for the magnetometer,
824the sensor should be rotated in a figure 8 motion until complete */
825int MPU9250::calibrateMag() {
826 // set the srd
827 if (setSrd(19) < 0) {
828 return -1;
829 }
830
831 // get a starting set of data
832 readSensor();
833 _hxmax = getMagX_uT();
834 _hxmin = getMagX_uT();
835 _hymax = getMagY_uT();
836 _hymin = getMagY_uT();
837 _hzmax = getMagZ_uT();
838 _hzmin = getMagZ_uT();
839
840 // collect data to find max / min in each channel
841 _counter = 0;
842 while (_counter < _maxCounts) {
843 _delta = 0.0f;
844 _framedelta = 0.0f;
845 readSensor();
846 _hxfilt = (_hxfilt*((float)_coeff-1)+(getMagX_uT()/_hxs+_hxb))/((float)_coeff);
847 _hyfilt = (_hyfilt*((float)_coeff-1)+(getMagY_uT()/_hys+_hyb))/((float)_coeff);
848 _hzfilt = (_hzfilt*((float)_coeff-1)+(getMagZ_uT()/_hzs+_hzb))/((float)_coeff);
849 if (_hxfilt > _hxmax) {
850 _delta = _hxfilt - _hxmax;
851 _hxmax = _hxfilt;
852 }
853 if (_delta > _framedelta) {
854 _framedelta = _delta;
855 }
856 if (_hyfilt > _hymax) {
857 _delta = _hyfilt - _hymax;
858 _hymax = _hyfilt;
859 }
860 if (_delta > _framedelta) {
861 _framedelta = _delta;
862 }
863 if (_hzfilt > _hzmax) {
864 _delta = _hzfilt - _hzmax;
865 _hzmax = _hzfilt;
866 }
867 if (_delta > _framedelta) {
868 _framedelta = _delta;
869 }
870 if (_hxfilt < _hxmin) {
871 _delta = abs(_hxfilt - _hxmin);
872 _hxmin = _hxfilt;
873 }
874 if (_delta > _framedelta) {
875 _framedelta = _delta;
876 }
877 if (_hyfilt < _hymin) {
878 _delta = abs(_hyfilt - _hymin);
879 _hymin = _hyfilt;
880 }
881 if (_delta > _framedelta) {
882 _framedelta = _delta;
883 }
884 if (_hzfilt < _hzmin) {
885 _delta = abs(_hzfilt - _hzmin);
886 _hzmin = _hzfilt;
887 }
888 if (_delta > _framedelta) {
889 _framedelta = _delta;
890 }
891 if (_framedelta > _deltaThresh) {
892 _counter = 0;
893 } else {
894 _counter++;
895 }
896 delay(20);
897 }
898
899 // find the magnetometer bias
900 _hxb = (_hxmax + _hxmin) / 2.0f;
901 _hyb = (_hymax + _hymin) / 2.0f;
902 _hzb = (_hzmax + _hzmin) / 2.0f;
903
904 // find the magnetometer scale factor
905 _hxs = (_hxmax - _hxmin) / 2.0f;
906 _hys = (_hymax - _hymin) / 2.0f;
907 _hzs = (_hzmax - _hzmin) / 2.0f;
908 _avgs = (_hxs + _hys + _hzs) / 3.0f;
909 _hxs = _avgs/_hxs;
910 _hys = _avgs/_hys;
911 _hzs = _avgs/_hzs;
912
913 // set the srd back to what it was
914 if (setSrd(_srd) < 0) {
915 return -2;
916 }
917 return 1;
918}
