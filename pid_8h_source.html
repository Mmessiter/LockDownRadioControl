11// This function reads the MPU6050 registers frequently to get the gyro and accelerometer signals
12void Read_MPU6050(void)
13{
14 Wire.beginTransmission(0x68);
15 Wire.write(0x1A);
16 Wire.write(0x05);
17 Wire.endTransmission();
18 Wire.beginTransmission(0x68);
19 Wire.write(0x1C);
20 Wire.write(0x10);
21 Wire.endTransmission();
22 Wire.beginTransmission(0x68);
23 Wire.write(0x3B);
24 Wire.endTransmission();
25 Wire.requestFrom(0x68, 6);
26 int16_t AccXLSB = Wire.read() << 8 | Wire.read();
27 int16_t AccYLSB = Wire.read() << 8 | Wire.read();
28 int16_t AccZLSB = Wire.read() << 8 | Wire.read();
29 Wire.beginTransmission(0x68);
30 Wire.write(0x1B);
31 Wire.write(0x8);
32 Wire.endTransmission();
33 Wire.beginTransmission(0x68);
34 Wire.write(0x43);
35 Wire.endTransmission();
36 Wire.requestFrom(0x68, 6);
37 int16_t GyroX = Wire.read() << 8 | Wire.read();
38 int16_t GyroY = Wire.read() << 8 | Wire.read();
39 int16_t GyroZ = Wire.read() << 8 | Wire.read();
40 RateRoll = (float)GyroX / 65.5;
41 RatePitch = (float)GyroY / 65.5;
42 RateYaw = (float)GyroZ / 65.5;
43 AccX = (float)AccXLSB / 4096;
44 AccY = (float)AccYLSB / 4096;
45 AccZ = (float)AccZLSB / 4096;
46 AngleRoll = atan(AccY / sqrt(AccX * AccX + AccZ * AccZ)) * 1 / (3.142 / 180);
47 AnglePitch = -atan(AccX / sqrt(AccY * AccY + AccZ * AccZ)) * 1 / (3.142 / 180);
48}

59// This function initialises the MPU6050 and calibrates the gyro
60
61void InitialiseTheMPU6050(){
62
63 int Iterations = 1000; // make bigger for better calibration after tests
64
65 Wire.setClock(400000);
66 Wire.begin();
67 delay(250);
68 Wire.beginTransmission(0x68);
69 Wire.write(0x6B);
70 Wire.write(0x00);
71 Wire.endTransmission();
72 for (int i = 0; i < Iterations; ++i)
73 {
74 Read_MPU6050();
75 RateCalibrationRoll += RateRoll;
76 RateCalibrationPitch += RatePitch;
77 RateCalibrationYaw += RateYaw;
78 delay(1);
79 BlinkFast();
80 }
81 RateCalibrationRoll /= Iterations;
82 RateCalibrationPitch /= Iterations;
83 RateCalibrationYaw /= Iterations;
84}

87// This function is the Kalman filter for 1D signals
88void Kalman_Filter_1d(float KalmanState, float KalmanUncertainty, float KalmanInput, float KalmanMeasurement)
89{
90 KalmanState = KalmanState + 0.004 * KalmanInput;
91 KalmanUncertainty = KalmanUncertainty + 0.004 * 0.004 * 4 * 4;
92 float KalmanGain = KalmanUncertainty * 1 / (1 * KalmanUncertainty + 3 * 3);
93 KalmanState = KalmanState + KalmanGain * (KalmanMeasurement - KalmanState);
94 KalmanUncertainty = (1 - KalmanGain) * KalmanUncertainty;
95 Kalman1DOutput[0] = KalmanState;
96 Kalman1DOutput[1] = KalmanUncertainty;
97}
