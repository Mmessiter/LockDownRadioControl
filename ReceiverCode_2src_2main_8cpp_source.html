202/************************************************************************************************************/
203bool ReadData()
204{
205 Connected = false;
206 while (CurrentRadio->available()) { // Get all, but use only the latest
207 LoadAckPayload();
208 Connected = true;
209 CurrentRadio->flush_tx(); // This avoids a lockup that happens when the FIFO gets full
210 CurrentRadio->writeAckPayload(1, &AckPayload, AckPayloadSize); // Send telemetry
211 delayMicroseconds(1500); // N.B. SOME DUFF NRF24L01 TRANSCEIVERS NEED THIS PAUSE. But not all.
212 CurrentRadio->read(&CompressedData, sizeof(CompressedData)); // Get Data
213 CurrentRadio->flush_rx(); // Flush FIFO to avoid a lock up
214 }
215 if (Connected) UseReceivedData();
216 return Connected;

317void ReadExtraParameters()
318{
319 uint16_t TwoBytes = 0;
320 uint8_t SwapWaveBand;
321
322 PacketNumber = ReceivedData[CHANNELSUSED];
323
324 switch (PacketNumber) {
325 case 0:
326 BindNow = ReceivedData[CHANNELSUSED + 2];
327 FailSafeSave = bool(ReceivedData[CHANNELSUSED + 1]);
328 if (FailSafeSave) {
329 TwoBytes = uint16_t(FS_byte2) + uint16_t(FS_byte1 << 8);
330 RebuildFlags(FailSafeChannel, TwoBytes);
331 }
332 break;
333 case 1:
334 FS_byte1 = ReceivedData[CHANNELSUSED + 1]; // These 2 bytes are 16 failsafe flags
335 FS_byte2 = ReceivedData[CHANNELSUSED + 2]; // These 2 bytes are 16 failsafe flags
336 break;
337 case 2:
338 Qnh = (ReceivedData[CHANNELSUSED + 1]) << 8; // 16 bits sent as two bytes for pressure here at sea level
339 Qnh += ReceivedData[CHANNELSUSED + 2];
340 if (OldQnh != Qnh) SendQnhToSensorHub();
341 OldQnh = Qnh; // Send new one once only
342 break;
343 case 3:
344 if ((ReceivedData[CHANNELSUSED + 2]) == 255) { // Mark this location
345 MarkHere();
346 ReceivedData[CHANNELSUSED + 2] = 0; // ... Once only
347 }
348 break;
349 case 4:
350 ModelMatched = ReceivedData[CHANNELSUSED + 1];
351 SwapWaveBand = ReceivedData[CHANNELSUSED + 2];
352 if (SwapWaveBand > 0) {
353 if (SwapWaveBand == 1) SetUKFrequencies();
354 if (SwapWaveBand == 2) SetTestFrequencies();
355 }
356 default:
357 break;
358 }
359 return;
360}

520// ******************************************************************************************************************************************************************
521FASTRUN void ReceiveData()
522{
523 uint32_t TimeTest;
524 if (Connected) {
525 if ((millis() - SensorHubAccessed) > 10) { // Reading Sensor hub 100 x per second should be enough
526 if (millis() - LastPacketArrivalTime < 1) { // If, and only if, we have still absolutely loads of time, do stuff now while waiting ...
527 SensorHubAccessed = millis(); // Note the moment of last attempted read.
528 if (!SensorHubDead) { // Better check it hasn't died.
529 TimeTest = millis(); // Time the I2C calls. If too long, don't repeat it ... save the model.
530 if (SensorHubConnected) ReadTheSensorHub(); // Sensor now has its own MCU. Calls return in far less that 6 ms unless it lost I2C synch
531 if (INA219Connected) INA219Volts = ina219.getBusVoltage_V(); // Get RX LIPO volts if connected separately (as will be needed on 'planes with no GPS fitted.)
532 if ((millis() - BootupMoment) > 5000) {
533 if ((millis() - TimeTest) > 6) SensorHubHasFailed(); // If sensor hub and/or INA219 fails, don't bother calling either again (It normally returns within 2 ms.
534 }
535 }
536 }
537 }
538 }
539 if (ReadData()) {
540 ReadExtraParameters(); // Check the extra parameters
541 }
542 else {
543 if (millis() - SBUSTimer >= SBUSRATE) { // No new packet yet - but maybe it's time to dispatch the last?
544 if (BoundFlag && (millis() > 10000)) {
545 if (Connected) {
546 KeepSbusHappy(); // if it's time - send a SBUS packet. It might be new data.
547 --SbusRepeats; // It's not really a "repeat".
548 }
549 }
550 }
551 if (millis() - LastPacketArrivalTime >= RECEIVE_TIMEOUT) {
552 Reconnect();
553 } // Try to reconnect.
554 }
555}

615
616union {uint64_t v64; uint8_t v8[8];} union1;
617union {uint64_t v64; uint8_t v8[8];} union2;
618
