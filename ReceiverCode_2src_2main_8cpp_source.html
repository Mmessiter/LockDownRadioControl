148 if (ServoCentrePulse[j] < 1000) {

150 }else{

152 }
153 analogWrite(PWMPins[j],GetPWMValue(ServoFrequency[j], S));
154 PreviousData[j] = ReceivedData[j];

337// ***************************************************************************************************************************************************
338// Here the GPS (Sensor) HUB is asked for 7 bytes of data over I2C.
339// The first IDLEN (=3) bytes are the ID (LAT, LNG, etc...)
340// The next 4 bytes are the value (as a float).
341// The ID changes with each call
342
343FASTRUN void ReadTheSensorHub()
344{
345
346#define IDLEN 3
347#define GPSI2CBYTES IDLEN + 4 // = 7 (only floats now)
348
349 char FIX[IDLEN + 1] = "FIX"; // GPS Fix
350 char SAT[IDLEN + 1] = "SAT"; // How many satellites
351 char LAT[IDLEN + 1] = "LAT"; // Latitude
352 char LON[IDLEN + 1] = "LON"; // Longitude
353 char ALT[IDLEN + 1] = "ALT"; // GPS Altitude
354 char SPD[IDLEN + 1] = "SPD"; // Speed
355 char COR[IDLEN + 1] = "COR"; // Course
356 char CTO[IDLEN + 1] = "CTO"; // Course to Mark
357 char DTO[IDLEN + 1] = "DTO"; // Distance to Mark
358 char HRS[IDLEN + 1] = "HRS"; // GMT Hours
359 char MNS[IDLEN + 1] = "MNS"; // GMT Minutes
360 char SEC[IDLEN + 1] = "SEC"; // GMT Seconds
361 char BLT[IDLEN + 1] = "BLT"; // Altitiude from BMP280
362 char TMP[IDLEN + 1] = "TMP"; // Temperature from BMP280
363 char VLT[IDLEN + 1] = "VLT"; // Volts from INA219
364 char DAY[IDLEN + 1] = "DAY"; // DAY
365 char MTH[IDLEN + 1] = "MTH"; // MONTH
366 char YER[IDLEN + 1] = "YER"; // YEAR
367 char RdataID[IDLEN + 1];
368 float RdataIn;
369 union
370 {
371 float Val32;
372 uint8_t Val8[4];
373 } Rdata; // 'union' allows access to every byte
374
375 Wire.requestFrom(SENSOR_HUB_I2C_ADDRESS, GPSI2CBYTES); // Ask hub for data
376 for (int j = 0; j < GPSI2CBYTES; ++j) {
377 if (Wire.available()) { // Listen to HUB
378 if (j < IDLEN) {
379 RdataID[j] = Wire.read(); // This gets the three-char data id (eg LAT)
380 }
381 else {
382 Rdata.Val8[j - IDLEN] = Wire.read(); // This gets the 64 bit value for that data ID
383 }
384 }
385 }
386 RdataID[3] = 0; // To terminate the ID string.
387 RdataIn = Rdata.Val32; // To re-assemble the 64 BIT data to a double
388
389 if (strcmp(FIX, RdataID) == 0) {
390 if (int(RdataIn) == 1) {
391 GpsFix = true;
392 return;
393 }
394 }
395 if (strcmp(LAT, RdataID) == 0) {
396 LatitudeGPS = RdataIn;
397 return;
398 }
399 if (strcmp(LON, RdataID) == 0) {
400 LongitudeGPS = RdataIn;
401 return;
402 }
403 if (strcmp(SPD, RdataID) == 0) {
404 SpeedGPS = RdataIn;
405 return;
406 }
407 if (strcmp(COR, RdataID) == 0) {
408 AngleGPS = RdataIn;
409 return;
410 }
411 if (strcmp(ALT, RdataID) == 0) {
412 AltitudeGPS = RdataIn;
413 return;
414 }
415 if (strcmp(DTO, RdataID) == 0) {
416 DistanceGPS = RdataIn;
417 return;
418 }
419 if (strcmp(SAT, RdataID) == 0) {
420 SatellitesGPS = uint8_t(RdataIn);
421 return;
422 }
423 if (strcmp(CTO, RdataID) == 0) {
424 CourseToGPS = RdataIn;
425 return;
426 }
427 if (strcmp(HRS, RdataID) == 0) {
428 HoursGPS = uint8_t(RdataIn);
429 return;
430 }
431 if (strcmp(MNS, RdataID) == 0) {
432 MinsGPS = uint8_t(RdataIn);
433 return;
434 }
435 if (strcmp(SEC, RdataID) == 0) {
436 SecsGPS = uint8_t(RdataIn);
437 return;
438 }
439 if (strcmp(BLT, RdataID) == 0) {
440 BaroAltitude = RdataIn;
441 return;
442 }
443 if (strcmp(TMP, RdataID) == 0) {
444 BaroTemperature = RdataIn;
445 return;
446 }
447 if (strcmp(VLT, RdataID) == 0) {
448 if (!INA219Connected) INA219Volts = RdataIn; // if there's a locally connected sensor, use it.
449 return;
450 }
451 if (strcmp(DAY, RdataID) == 0) {
452 DayGPS = uint8_t(RdataIn);
453 return;
454 }
455 if (strcmp(MTH, RdataID) == 0) {
456 MonthGPS = uint8_t(RdataIn);
457 return;
458 }
459 if (strcmp(YER, RdataID) == 0) {
460 YearGPS = uint8_t(RdataIn);
461 return;
462 }
463}
