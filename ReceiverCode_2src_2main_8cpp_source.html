312// Here the GPS (Sensor) HUB is asked for 7 bytes of data over I2C.
313// The first IDLEN (=3) bytes are the ID (LAT, LNG, etc...)
314// The next 4 bytes are the value (as a float).
315// The ID changes with each call
316
317FASTRUN void ReadTheSensorHub()
318{
319
320#define IDLEN 3
321#define GPSI2CBYTES IDLEN + 4 // = 7 (only floats now)
322
323 char FIX[IDLEN + 1] = "FIX"; // GPS Fix
324 char SAT[IDLEN + 1] = "SAT"; // How many satellites
325 char LAT[IDLEN + 1] = "LAT"; // Latitude
326 char LON[IDLEN + 1] = "LON"; // Longitude
327 char ALT[IDLEN + 1] = "ALT"; // GPS Altitude
328 char SPD[IDLEN + 1] = "SPD"; // Speed
329 char COR[IDLEN + 1] = "COR"; // Course
330 char CTO[IDLEN + 1] = "CTO"; // Course to Mark
331 char DTO[IDLEN + 1] = "DTO"; // Distance to Mark
332 char HRS[IDLEN + 1] = "HRS"; // GMT Hours
333 char MNS[IDLEN + 1] = "MNS"; // GMT Minutes
334 char SEC[IDLEN + 1] = "SEC"; // GMT Seconds
335 char BLT[IDLEN + 1] = "BLT"; // Altitiude from BMP280
336 char TMP[IDLEN + 1] = "TMP"; // Temperature from BMP280
337 char VLT[IDLEN + 1] = "VLT"; // Volts from INA219
338 char DAY[IDLEN + 1] = "DAY"; // DAY
339 char MTH[IDLEN + 1] = "MTH"; // MONTH
340 char YER[IDLEN + 1] = "YER"; // YEAR
341 char RdataID[IDLEN + 1];
342 float RdataIn;
343 union
344 {
345 float Val32;
346 uint8_t Val8[4];
347 } Rdata; // 'union' allows access to every byte
348
349 Wire.requestFrom(SENSOR_HUB_I2C_ADDRESS, GPSI2CBYTES); // Ask hub for data
350 for (int j = 0; j < GPSI2CBYTES; ++j) {
351 if (Wire.available()) { // Listen to HUB
352 if (j < IDLEN) {
353 RdataID[j] = Wire.read(); // This gets the three-char data id (eg LAT)
354 }
355 else {
356 Rdata.Val8[j - IDLEN] = Wire.read(); // This gets the 64 bit value for that data ID
357 }
358 }
359 }
360 RdataID[3] = 0; // To terminate the ID string.
361 RdataIn = Rdata.Val32; // To re-assemble the 64 BIT data to a double
362
363 if (strcmp(FIX, RdataID) == 0) {
364 if (int(RdataIn) == 1) {
365 GpsFix = true;
366 return;
367 }
368 }
369 if (strcmp(LAT, RdataID) == 0) {
370 LatitudeGPS = RdataIn;
371 return;
372 }
373 if (strcmp(LON, RdataID) == 0) {
374 LongitudeGPS = RdataIn;
375 return;
376 }
377 if (strcmp(SPD, RdataID) == 0) {
378 SpeedGPS = RdataIn;
379 return;
380 }
381 if (strcmp(COR, RdataID) == 0) {
382 AngleGPS = RdataIn;
383 return;
384 }
385 if (strcmp(ALT, RdataID) == 0) {
386 AltitudeGPS = RdataIn;
387 return;
388 }
389 if (strcmp(DTO, RdataID) == 0) {
390 DistanceGPS = RdataIn;
391 return;
392 }
393 if (strcmp(SAT, RdataID) == 0) {
394 SatellitesGPS = uint8_t(RdataIn);
395 return;
396 }
397 if (strcmp(CTO, RdataID) == 0) {
398 CourseToGPS = RdataIn;
399 return;
400 }
401 if (strcmp(HRS, RdataID) == 0) {
402 HoursGPS = uint8_t(RdataIn);
403 return;
404 }
405 if (strcmp(MNS, RdataID) == 0) {
406 MinsGPS = uint8_t(RdataIn);
407 return;
408 }
409 if (strcmp(SEC, RdataID) == 0) {
410 SecsGPS = uint8_t(RdataIn);
411 return;
412 }
413 if (strcmp(BLT, RdataID) == 0) {
414 BaroAltitude = RdataIn;
415 return;
416 }
417 if (strcmp(TMP, RdataID) == 0) {
418 BaroTemperature = RdataIn;
419 return;
420 }
421 if (strcmp(VLT, RdataID) == 0) {
422 if (!INA219Connected) INA219Volts = RdataIn; // if there's a locally connected sensor, use it.
423 return;
424 }
425 if (strcmp(DAY, RdataID) == 0) {
426 DayGPS = uint8_t(RdataIn);
427 return;
428 }
429 if (strcmp(MTH, RdataID) == 0) {
430 MonthGPS = uint8_t(RdataIn);
431 return;
432 }
433 if (strcmp(YER, RdataID) == 0) {
434 YearGPS = uint8_t(RdataIn);
435 return;
436 }
437}

523void teensyMAC(uint8_t* mac)
524{ // GET UNIQUE TEENSY 4.0 ID
525 for (uint8_t by = 0; by < 2; by++) mac[by] = (HW_OCOTP_MAC1 >> ((1 - by) * 8)) & 0xFF;
526 for (uint8_t by = 0; by < 4; by++) mac[by + 2] = (HW_OCOTP_MAC0 >> ((3 - by) * 8)) & 0xFF;
