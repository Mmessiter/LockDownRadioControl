97 int16_t Adafruit_INA219::getCurrent_raw() {
98  uint16_t value;
99 
100  // Sometimes a sharp load will reset the INA219, which will
101  // reset the cal register, meaning CURRENT and POWER will
102  // not be available ... avoid this by always setting a cal
103  // value even if it's an unfortunate extra step
104  Adafruit_BusIO_Register calibration_reg =
105  Adafruit_BusIO_Register(i2c_dev, INA219_REG_CALIBRATION, 2, MSBFIRST);
106  calibration_reg.write(ina219_calValue, 2);
107 
108  // Now we can safely read the CURRENT register!
109  Adafruit_BusIO_Register current_reg =
110  Adafruit_BusIO_Register(i2c_dev, INA219_REG_CURRENT, 2, MSBFIRST);
111  current_reg.read(&value);
112  return value;
113 }

185 void Adafruit_INA219::setCalibration_32V_2A() {
186  // By default we use a pretty huge range for the input voltage,
187  // which probably isn't the most appropriate choice for system
188  // that don't use a lot of power. But all of the calculations
189  // are shown below if you want to change the settings. You will
190  // also need to change any relevant register settings, such as
191  // setting the VBUS_MAX to 16V instead of 32V, etc.
192 
193  // VBUS_MAX = 32V (Assumes 32V, can also be set to 16V)
194  // VSHUNT_MAX = 0.32 (Assumes Gain 8, 320mV, can also be 0.16, 0.08,
195  // 0.04) RSHUNT = 0.1 (Resistor value in ohms)
196 
197  // 1. Determine max possible current
198  // MaxPossible_I = VSHUNT_MAX / RSHUNT
199  // MaxPossible_I = 3.2A
200 
201  // 2. Determine max expected current
202  // MaxExpected_I = 2.0A
203 
204  // 3. Calculate possible range of LSBs (Min = 15-bit, Max = 12-bit)
205  // MinimumLSB = MaxExpected_I/32767
206  // MinimumLSB = 0.000061 (61uA per bit)
207  // MaximumLSB = MaxExpected_I/4096
208  // MaximumLSB = 0,000488 (488uA per bit)
209 
210  // 4. Choose an LSB between the min and max values
211  // (Preferrably a roundish number close to MinLSB)
212  // CurrentLSB = 0.0001 (100uA per bit)
213 
214  // 5. Compute the calibration register
215  // Cal = trunc (0.04096 / (Current_LSB * RSHUNT))
216  // Cal = 4096 (0x1000)
217 
218  ina219_calValue = 4096;
219 
220  // 6. Calculate the power LSB
221  // PowerLSB = 20 * CurrentLSB
222  // PowerLSB = 0.002 (2mW per bit)
223 
224  // 7. Compute the maximum current and shunt voltage values before overflow
225  //
226  // Max_Current = Current_LSB * 32767
227  // Max_Current = 3.2767A before overflow
228  //
229  // If Max_Current > Max_Possible_I then
230  // Max_Current_Before_Overflow = MaxPossible_I
231  // Else
232  // Max_Current_Before_Overflow = Max_Current
233  // End If
234  //
235  // Max_ShuntVoltage = Max_Current_Before_Overflow * RSHUNT
236  // Max_ShuntVoltage = 0.32V
237  //
238  // If Max_ShuntVoltage >= VSHUNT_MAX
239  // Max_ShuntVoltage_Before_Overflow = VSHUNT_MAX
240  // Else
241  // Max_ShuntVoltage_Before_Overflow = Max_ShuntVoltage
242  // End If
243 
244  // 8. Compute the Maximum Power
245  // MaximumPower = Max_Current_Before_Overflow * VBUS_MAX
246  // MaximumPower = 3.2 * 32V
247  // MaximumPower = 102.4W
248 
249  // Set multipliers to convert raw current/power values
250  ina219_currentDivider_mA = 10; // Current LSB = 100uA per bit (1000/100 = 10)
251  ina219_powerMultiplier_mW = 2; // Power LSB = 1mW per bit (2/1)
252 
253  // Set Calibration register to 'Cal' calculated above
254  Adafruit_BusIO_Register calibration_reg =
255  Adafruit_BusIO_Register(i2c_dev, INA219_REG_CALIBRATION, 2, MSBFIRST);
256  calibration_reg.write(ina219_calValue, 2);
257 
258  // Set Config register to take into account the settings above
259  uint16_t config = INA219_CONFIG_BVOLTAGERANGE_32V |
260  INA219_CONFIG_GAIN_8_320MV | INA219_CONFIG_BADCRES_12BIT |
261  INA219_CONFIG_SADCRES_12BIT_1S_532US |
262  INA219_CONFIG_MODE_SANDBVOLT_CONTINUOUS;
263  Adafruit_BusIO_Register config_reg =
264  Adafruit_BusIO_Register(i2c_dev, INA219_REG_CONFIG, 2, MSBFIRST);
265  config_reg.write(config, 2);
266 }

273 void Adafruit_INA219::powerSave(bool on) {
