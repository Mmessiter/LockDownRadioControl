76 void AnalogBufferDMA::init(ADC *adc, int8_t adc_num)
77 {
78  // enable DMA and interrupts
79 #ifdef DEBUG_DUMP_DATA
80  Serial.println("AnalogBufferDMA::init");
81  Serial.flush();
82 #endif
83 
84 #ifndef KINETISL
85  // setup a DMA Channel.
86  // Now lets see the different things that RingbufferDMA setup for us before
87  // See if we were created with one or two buffers. If one assume we stop on completion, else assume continuous.
88  if (_buffer2 && _buffer2_count)
89  {
90 #ifdef ADC_DUAL_ADCS
91  _dmasettings_adc[0].source((volatile uint16_t &)((adc_num == 1) ? SOURCE_ADC_1 : SOURCE_ADC_0));
92 #else
93  _dmasettings_adc[0].source((volatile uint16_t &)(SOURCE_ADC_0));
94 #endif
95  _dmasettings_adc[0].destinationBuffer((uint16_t *)_buffer1, _buffer1_count * 2); // 2*b_size is necessary for some reason
96  _dmasettings_adc[0].replaceSettingsOnCompletion(_dmasettings_adc[1]); // go off and use second one...
97  _dmasettings_adc[0].interruptAtCompletion(); //interruptAtHalf or interruptAtCompletion
98 #ifdef ADC_DUAL_ADCS
99  _dmasettings_adc[1].source((volatile uint16_t &)((adc_num == 1) ? SOURCE_ADC_1 : SOURCE_ADC_0));
100  _dmasettings_adc[1].destinationBuffer((uint16_t *)_buffer2, _buffer2_count * 2); // 2*b_size is necessary for some reason
101  _dmasettings_adc[1].replaceSettingsOnCompletion(_dmasettings_adc[0]); // Cycle back to the first one
102  _dmasettings_adc[1].interruptAtCompletion(); //interruptAtHalf or interruptAtCompletion
103 #endif
104 
105  _dmachannel_adc = _dmasettings_adc[0];
106 
107  _stop_on_completion = false;
108  }
109  else
110  {
111 // Only one buffer so lets just setup the dmachannel ...
112 // Serial.printf("AnalogBufferDMA::init Single buffer %d\n", adc_num);
113 #ifdef ADC_DUAL_ADCS
114  _dmachannel_adc.source((volatile uint16_t &)((adc_num == 1) ? SOURCE_ADC_1 : SOURCE_ADC_0));
115 #else
116  _dmachannel_adc.source((volatile uint16_t &)(SOURCE_ADC_0));
117 #endif
118  _dmachannel_adc.destinationBuffer((uint16_t *)_buffer1, _buffer1_count * 2); // 2*b_size is necessary for some reason
119  _dmachannel_adc.interruptAtCompletion(); //interruptAtHalf or interruptAtCompletion
120  _dmachannel_adc.disableOnCompletion(); // we will disable on completion.
121  _stop_on_completion = true;
122  }
123 
124  if (adc_num == 1)
125  {
126 #ifdef ADC_DUAL_ADCS
127  _activeObjectPerADC[1] = this;
128  _dmachannel_adc.attachInterrupt(&adc_1_dmaISR);
129  _dmachannel_adc.triggerAtHardwareEvent(DMAMUX_ADC_1); // start DMA channel when ADC finishes a conversion
130 #endif
131  }
132  else
133  {
134  _activeObjectPerADC[0] = this;
135  _dmachannel_adc.attachInterrupt(&adc_0_dmaISR);
136  _dmachannel_adc.triggerAtHardwareEvent(DMAMUX_ADC_0); // start DMA channel when ADC finishes a conversion
137  }
138  //arm_dcache_flush((void*)dmaChannel, sizeof(dmaChannel));
139  _dmachannel_adc.enable();
140 
141  adc->adc[adc_num]->continuousMode();
142  adc->adc[adc_num]->enableDMA();
143 
144 #ifdef DEBUG_DUMP_DATA
145  dumpDMA_TCD(&_dmachannel_adc);
146  dumpDMA_TCD(&_dmasettings_adc[0]);
147  dumpDMA_TCD(&_dmasettings_adc[1]);
148 #if defined(__IMXRT1062__) // Teensy 4.0
149 
150  if (adc_num == 1)
151  {
152  Serial.printf("ADC2: HC0:%x HS:%x CFG:%x GC:%x GS:%x\n", ADC2_HC0, ADC2_HS, ADC2_CFG, ADC2_GC, ADC2_GS);
153  }
154  else
155  {
156  Serial.printf("ADC1: HC0:%x HS:%x CFG:%x GC:%x GS:%x\n", ADC1_HC0, ADC1_HS, ADC1_CFG, ADC1_GC, ADC1_GS);
157  }
158 #endif
159 #endif
160 #else
161  // Kinetisl (TLC)
162  // setup a DMA Channel.
163  // Now lets see the different things that RingbufferDMA setup for us before
164  _dmachannel_adc.source((volatile uint16_t &)(SOURCE_ADC_0));
165  ;
166  _dmachannel_adc.destinationBuffer((uint16_t *)_buffer1, _buffer1_count * 2); // 2*b_size is necessary for some reason
167  _dmachannel_adc.disableOnCompletion(); // ISR will hae to restart with other buffer
168  _dmachannel_adc.interruptAtCompletion(); //interruptAtHalf or interruptAtCompletion
169  _activeObjectPerADC[0] = this;
170  _dmachannel_adc.attachInterrupt(&adc_0_dmaISR);
171  _dmachannel_adc.triggerAtHardwareEvent(DMAMUX_ADC_0); // start DMA channel when ADC finishes a conversion
172  _dmachannel_adc.enable();
173 
174  adc->startContinuous(adc_num);
175  adc->adc[adc_num]->enableDMA();
176 #ifdef DEBUG_DUMP_DATA
177  dumpDMA_TCD(&_dmachannel_adc);
178 #endif
179 
180 #endif
181 
182  _last_isr_time = millis();
183 }

185 //=============================================================================
186 // stopOnCompletion: allows you to turn on or off stopping when a DMA buffer
187 // has completed filling. Default is on when only one buffer passed in to the
188 // constructor and off if two buffers passed in.
189 //=============================================================================
190 void AnalogBufferDMA::stopOnCompletion(bool stop_on_complete)
191 {
192 #ifndef KINETISL
193  if (stop_on_complete)
194  _dmachannel_adc.TCD->CSR |= DMA_TCD_CSR_DREQ;
195  else
196  _dmachannel_adc.TCD->CSR &= ~DMA_TCD_CSR_DREQ;
197 #else
198  if (stop_on_complete)
199  _dmachannel_adc.CFG->DCR |= DMA_DCR_D_REQ;
200  else
201  _dmachannel_adc.CFG->DCR &= ~DMA_DCR_D_REQ;
202 #endif
203  _stop_on_completion = stop_on_complete;
204 }
205 
206 //=============================================================================
207 // ClearCompletion: if we have stop on completion, then clear the completion state
208 // i.e. reenable the dma operation. Note only valid if we are
209 // in the stopOnCompletion state.
210 //=============================================================================
211 bool AnalogBufferDMA::clearCompletion()
212 {
213  if (!_stop_on_completion)
214  return false;
215  // should probably check to see if we are dsiable or not...
216  _dmachannel_adc.enable();
217  return true;
218 }

220 //=============================================================================
221 // processADC_DMAISR: Process the DMA completion ISR
222 // common for both ISRs on those processors who have more than one ADC
223 //=============================================================================
224 void AnalogBufferDMA::processADC_DMAISR()
225 {
226  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN));
227  uint32_t cur_time = millis();
228 
229  _interrupt_count++;
230  _interrupt_delta_time = cur_time - _last_isr_time;
231  _last_isr_time = cur_time;
232  // update the internal buffer positions
233  _dmachannel_adc.clearInterrupt();
234 #ifdef KINETISL
235  // Lets try to clear the previous interrupt, change buffers
236  // and restart
237  if (_buffer2 && (_interrupt_count & 1))
238  {
239  _dmachannel_adc.destinationBuffer((uint16_t *)_buffer2, _buffer2_count * 2); // 2*b_size is necessary for some reason
240  }
241  else
242  {
243  _dmachannel_adc.destinationBuffer((uint16_t *)_buffer1, _buffer1_count * 2); // 2*b_size is necessary for some reason
244  }
245 
246  // If we are not stopping on completion, then reenable...
247  if (!_stop_on_completion)
248  _dmachannel_adc.enable();
249 
250 #endif
251 }
