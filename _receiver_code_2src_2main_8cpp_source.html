192 bool ReadData()
193 {
194  uint16_t CompressedData[COMPRESSEDWORDS]; // 30 bytes -> 40 bytes when uncompressed
195  Connected = false;
196  if (CurrentRadio->available()) {
197  LoadAckPayload();
198  Connected = true;
199  CurrentRadio->flush_tx(); // This avoids a lockup that happens when the FIFO gets full.**************
200  CurrentRadio->writeAckPayload(1, &AckPayload, AckPayloadSize); // Send telemetry (actual length plus 0)
201  CurrentRadio->read(&CompressedData, sizeof(CompressedData)); // Get Data
202  Decompress(ReceivedData, CompressedData, UNCOMPRESSEDWORDS); // decompress data
203  MapToSBUS();
204  CurrentRadio->flush_rx(); // This avoids a lockup that happens when the FIFO gets full. **************
205  ClearAckPayload();
206  LastConnectionMoment = millis();
207  if (HopNow) { // this flag gets set in LoadAckPayload();
208  FailSafeDataLoaded = false; // Ack payload instructed to Hop at next opportunity...
209  HopToNextFrequency(); // So hop now
210  HopNow = false; // and clear the flag.
211  DoSensors(); // read sensors while TX hops an catches up
212  }
213  }
214  return Connected;
