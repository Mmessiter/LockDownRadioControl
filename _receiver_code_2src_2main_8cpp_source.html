103 FASTRUN double HowFar(double latitude_new, double longitude_new, double latitude_old, double longitude_old) {
104  double RadiusOfTheEarth = 6372797.56085; // Meters by the way
105  double DegreesToRadians = 3.14159265358979323846 / 180;
106  double lat_new = latitude_old * DegreesToRadians;
107  double lat_old = latitude_new * DegreesToRadians;
108  double lat_diff = (latitude_new-latitude_old) * DegreesToRadians;
109  double lng_diff = (longitude_new-longitude_old) * DegreesToRadians;
110  double a = sin(lat_diff/2) * sin(lat_diff/2) + cos(lat_new) * cos(lat_old) * sin(lng_diff/2) * sin(lng_diff/2);
111  double c = 2 * atan2(sqrt(a), sqrt(1-a));
112  double distance = RadiusOfTheEarth * c;
113  return distance;
114  }

241  Connected = false;

246  CurrentRadio->writeAckPayload(1, &AckPayload, AckPayloadSize); // Send telemetry (actual length plus 0)
247  CurrentRadio->read(&CompressedData, sizeof(CompressedData)); // Get Data
248  Decompress(ReceivedData, CompressedData, UNCOMPRESSEDWORDS); // decompress data
249  MapToSBUS();
250  CurrentRadio->flush_rx(); // This avoids a lockup that happens when the FIFO gets full. **************

313 void CheckParams()
314 { uint16_t TwoBytes = 0;
315  PacketNumber = ReceivedData[CHANNELSUSED];
316  switch (PacketNumber) {
317  case 0:
318  BindNow = ReceivedData[CHANNELSUSED + 2];
319  FailSafeSave = bool(ReceivedData[CHANNELSUSED + 1]);
320  if (FailSafeSave) {
321  TwoBytes = uint16_t(FS_byte2) + uint16_t(FS_byte1 << 8);
322  RebuildFlags(FailSafeChannel, TwoBytes);
323  }
324  break;
325  case 1:
326  FS_byte1 = ReceivedData[CHANNELSUSED + 1]; // These 2 bytes are 16 failsafe flags
327  FS_byte2 = ReceivedData[CHANNELSUSED + 2]; // These 2 bytes are 16 failsafe flags
328  break;
329  case 2:
330  Qnh = (ReceivedData[CHANNELSUSED + 1]) << 8; // 16 bits sent as two bytes for pressure here at sea level
331  Qnh += ReceivedData[CHANNELSUSED + 2];
332  default:
333  break;
334  }
335  return;
