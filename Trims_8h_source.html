197void CheckHardwareTrims()
198{
199 if ((millis() - TrimTimer) < TrimRepeatSpeed) return; // check occasionally for trim press
200 TrimTimer = millis();
201 for (int i = 0; i < 8; ++i) {
202 if (TrimSwitch[i]) {
203 if (DefiningTrims) {
204 SetATrimDefinition(i);
205 return;
206 }
207 MoveaTrim(i);
208 TransmitterLastManaged = 0; // to speed up repeat
209 TrimRepeatSpeed -= (TrimRepeatSpeed / 4); // accelerate repeat...
210 if (TrimRepeatSpeed < 10) TrimRepeatSpeed = 30; // ... up to a point...
211 }
212 }
213}

217int GetTrimAmount(uint8_t InputChannel)
218{
219 int tt = InputChannel;
220 if (SticksMode == 2) {
221 if (InputChannel == 1) tt = 2;
222 if (InputChannel == 2) tt = 1;
223 }
224 int TrimAmount = (Trims[Bank][tt] - 80) * TrimMultiplier; // TRIMS on lower four input channels (80 is mid point !! (range 40 - 80 - 120))
225
226 if ((tt == 1) && (SticksMode == 2)) { // bug fix for mode 2 throttle trim - it was reversed
227 TrimAmount = 80 - TrimAmount;
228 }
229 if ((tt == 2) && (SticksMode == 1)) { // bug fix for mode 1 throttle trim - it was reversed
230 TrimAmount = 80 - TrimAmount;
231 }
232
233 return TrimAmount;
234}

237void UpdateTrimView()
238{

240 uint8_t p;
241 char TrimViewChannels[4][4] = {"ch1", "ch4", "ch2", "ch3"};
242 char TrimViewNumbers[4][3] = {"n1", "n4", "n2", "n3"};
243 char TrimChannelNames[4][3] = {"c1", "c2", "c3", "c4"};
244
245 if (CurrentView == FRONTVIEW || (CurrentView == TRIM_VIEW)) {
246 for (int i = 0; i < 4; ++i) {
247 p = i;
248 if (SticksMode == 2) {
249 if (i == 1) p = 2;
250 if (i == 2) p = 1;
251 }
252 uint8_t pp = InputTrim[p];
253 SendValue(TrimViewChannels[p], (Trims[Bank][pp]));
254 SendValue(TrimViewNumbers[p], (Trims[Bank][pp] - 80));
255 if (CurrentView == TRIM_VIEW) SendText(TrimChannelNames[i], ChannelNames[pp]);
256 }
257 }
258}
