201 TrimTimer = millis();

331{

334 if (Trims[Bank][t] >= 120) {
335 Trims[Bank][t] = 120;

337

342 }

381void MoveaTrim(uint8_t i)
382{
383 uint8_t Elevator = 1;
384 uint8_t Throttle = 2;
385
386 if (SticksMode == 2) {
387 Elevator = 2;
388 Throttle = 1;
389 }
390
391 switch (i) {
392 case 0:
393 IncTrim(InputTrim[0]); // Aileron
394 break;
395 case 1:
396 DecTrim(InputTrim[0]); // Aileron
397 break;
398 case 2:
399 IncTrim(InputTrim[Elevator]);
400 break;
401 case 3:
402 DecTrim(InputTrim[Elevator]);
403 break;
404 case 4:
405 DecTrim(InputTrim[Throttle]);
406 break;
407 case 5:
408 IncTrim(InputTrim[Throttle]);
409 break;
410 case 6:
411 IncTrim(InputTrim[3]); // Rudder
412 break;
413 case 7:
414 DecTrim(InputTrim[3]); // Rudder
415 break;
416 default:
417 break;
418 }
419 if (CopyTrimsToAll) {
420 for (i = 0; i < 4; ++i) {
421 for (int fm = 1; fm < 5; ++fm) {
422 Trims[fm][i] = Trims[Bank][i];
423 }
424 }
425 }

429void TrimsToSubtrim(){ // Store trims to subtrims and reset trims to centre
430
431 // Trims range from 40 to 120. 80 is centre.
432
433 char Prompt[] = "Move these trims to Subtrims?\r\n\r\nThis would re-centre these trims.";
434 char ItsCancelled[] = "Cancelled.\r\nSubtrims were not changed.\r\nLook at the Subtrim view \r\nif you'd like to check.";
435 char ItsAllDone[] = "Done!\r\nSubtrims have been amended.\r\n\r\nLook at the Subtrim view \r\nif you'd like to check.";
436
437 if (GetConfirmation(pTrimView,Prompt)) {
438 for (int i = 0; i < 4; ++i) {
439 int temp = (int) (Trims[Bank][InputTrim[i]] - 80) * (int) TrimMultiplier; // 'int temp' is needed to avoid overflow
440 temp /= 5; // Subtrims' multiplier is 5 always
441 if (temp + SubTrims[InputTrim[i]] > 255) temp = 255 - SubTrims[InputTrim[i]]; // Avoid overflow
442 SubTrims[InputTrim[i]] += temp; // Trims *affected* channels are in the InputTrim[] array
443 if (SubTrims[i] > 255) SubTrims[i] = 255; // Avoid overflow
444 for (int j = 1; j < 5; ++j) Trims[j][InputTrim[i]] = 80; // Reset all four normal trims to centre
445 }
446 PlaySound(BEEPCOMPLETE);
447 MsgBox(pTrimView,ItsAllDone); // Done!
448 ForceDataRedisplay(); // force Update trim view
449 UpdateTrimView(); // Update trim view
450 SaveOneModel(ModelNumber); // Save model memory
451 }else{
452 MsgBox(pTrimView,ItsCancelled); // Cancelled
453 ForceDataRedisplay(); // force Update trim view
454 UpdateTrimView(); // Update trim view
455 }
456}
