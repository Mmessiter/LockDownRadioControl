39// This function converts an int to a char[] array, then adds a comma, a dot, or nothing at the end.
40// It builds the char[] array at a pointer (*s) Where there MUST be enough space for all characters plus a zero terminator.
41// It dates for a very early time when I didn't know about standard library functions!
42// But it works just fine, so it says in.
43
44FASTRUN char* Str(char* s, int n, int comma) // comma = 0 for nothing, 1 for a comma, 2 for a dot.
45{
46 int r, i, m, flag;
47 char cma[] = ",";
48 char dot[] = ".";
49
50 flag = 0;
51 i = 0;
52 m = 1000000000;
53 if (n < 0) {
54 s[0] = '-';
55 i = 1;
56 n = -n;
57 }
58 if (n == 0) {
59 s[0] = 48;
60 s[1] = 0;
61
62 if (comma == 1) {
63 strcat(s, cma);
64 }
65 if (comma == 2) {
66 strcat(s, dot);
67 }
68 return s;
69 }
70 while (m >= 1) {
71 r = n / m;
72 if (r > 0) {
73 flag = 1;
74 } // first digit
75 if (flag == 1) {
76 s[i] = 48 + r;
77 ++i;
78 s[i] = 0;
79 }
80 n -= (r * m);
81 m /= 10;
82 }
83 if (comma == 1) {
84 strcat(s, cma);
85 }
86 if (comma == 2) {
87 strcat(s, dot);
88 }
89 return s;
90}

156 DeltaGMT ^= 0xffff; // toggle every bit! :-)

186 Wire.write(decToBcd(GmonthDay));

190 Wire.endTransmission();
191}

209void AdjustDateTime(uint8_t MinChange, uint8_t HourChange, uint8_t YearChange, uint8_t MonthChange, uint8_t DateChange)
210{
211 ReadTheRTC();
212 Gminute += MinChange;
213 if (Gminute > 59) {
214 Gminute = 0;
215 if (Ghour < 23) {
216 ++Ghour;
217 }
218 }
219 if (Gminute < 1) {
220 Gminute = 0;
221 if (Ghour > 0) {
222 --Ghour;
223 }
224 }
225 Ghour += HourChange;
226 if (Ghour < 0) Ghour = 0;
227 if (Ghour > 23) Ghour = 23;
228 Gyear += YearChange;
229 if (Gyear < 0) Gyear = 0;
230 if (Gyear > 99) Gyear = 99;
231 Gmonth += MonthChange;
232 if (Gmonth < 1) Gmonth = 1;
233 if (Gmonth > 12) Gmonth = 12;
234 GmonthDay += DateChange;
235 if (GmonthDay < 1) GmonthDay = 1;
236 if (GmonthDay > 31) GmonthDay = 31;
237 SetTheRTC();
238}

382bool getDate(const char* str)
383{
384 char Month[12];
385 int Day, Year;
386 uint8_t monthIndex;
387 const char* monthName[12] = {
388 "Jan", "Feb", "Mar", "Apr", "May", "Jun",
389 "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
390 if (sscanf(str, "%s %d %d", Month, &Day, &Year) != 3) return false;
391 for (monthIndex = 0; monthIndex < 12; ++monthIndex) {
392 if (strcmp(Month, monthName[monthIndex]) == 0) break;
393 }
394 if (monthIndex >= 12) return false;
395 tm.Day = Day;
396 tm.Month = monthIndex + 1;
397 tm.Year = CalendarYrToTm(Year);
398 return true;
399}

413void ReadTime()
414{
415
416 static char month[12][15] = {"January", "February", "March", "April", "May", "June", "July", "August", "Sept", "October", "November", "December"};
417 static char ShortMonth[12][7] = {"Jan. ", "Feb. ", "Mar. ", "Apr. ", "May ", "June ", "July ", "Aug. ", "Sept ", "Oct. ", "Nov. ", "Dec. "};
418
419 char NB[10];
420 char TimeString[80];
421 char Space[] = " ";
422 char colon[] = ":";
423 char colon1[] = ".";
424 char zero[] = "0";
425 char DateTime[] = "DateTime";
426
427 uint8_t DisplayedHour;
428 FixDeltaGMTSign();
429 if (CurrentView == FRONTVIEW || CurrentView == OPTIONVIEW2) {
430 if (RTC.read(tm)) {
431 strcpy(TimeString, Str(NB, tm.Day + DateFix, 0));
432 if (CurrentView == OPTIONVIEW2)
433 {
434 if ((tm.Day) < 10) {
435 strcat(TimeString, Space); // to align better the rest of the data
436 strcat(TimeString, Space);
437 }
438 }
439 strcat(TimeString, Space);
440 if (CurrentView == OPTIONVIEW2)
441 {
442 strcat(TimeString, ShortMonth[tm.Month - 1]);
443 }
444 else
445 {
446 strcat(TimeString, month[tm.Month - 1]);
447 }
448 strcat(TimeString, Space);
449 strcat(TimeString, (Str(NB, tmYearToCalendar(tm.Year), 0)));
450 strcat(TimeString, Space);
451 DisplayedHour = tm.Hour + DeltaGMT;
452 DateFix = 0;
453 if (DisplayedHour > 24) {
454 DisplayedHour -= 24;
455 DateFix = 1;
456 }
457 if (DisplayedHour < 0) {
458 DisplayedHour += 24;
459 DateFix = -1;
460 }
461 if (MayBeAddZero(DisplayedHour)) strcat(TimeString, zero);
462 strcat(TimeString, Str(NB, DisplayedHour, 0));
463 if (FHSS_data::UkRules) strcat(TimeString, colon);
464 if (!FHSS_data::UkRules) strcat(TimeString, colon1);
465 if (MayBeAddZero(tm.Minute)) strcat(TimeString, zero);
466 strcat(TimeString, Str(NB, tm.Minute, 0));
467 if (FHSS_data::UkRules) strcat(TimeString, colon);
468 if (!FHSS_data::UkRules) strcat(TimeString, colon1);
469 if (MayBeAddZero(tm.Second)) strcat(TimeString, zero);
470 strcat(TimeString, Str(NB, tm.Second, 0));
471 SendText(DateTime, TimeString);
472 }
473 }

595{

602 ++j;
603 text2[j] = 0;

612 j = (text2[0] - 48) * 10;
613 j += (text2[1] - 48);

850/*********************************************************************************************************************************/
851int GetSuccessRate()
852{
853 uint16_t Total = 0;
854 uint16_t SuccessRate;
855 uint16_t Perfection = (PERFECTPACKETSPERSECOND * (uint16_t)ConnectionAssessSeconds);
856
857 for (uint16_t i = 0; i < Perfection; ++i) { // PERFECTPACKETSPERSECOND (126) packets per second are either good or bad
858 Total += PacketsHistoryBuffer[i];
859 }
860 Total += (Perfection - Total) / 2; // about half made it but were simply unacknowledged
861 SuccessRate = (Total * 100) / Perfection; // return a percentage of total good packets
862 return SuccessRate;
863}

905FASTRUN float MapWithExponential(float xx, float Xxmin, float Xxmax, float Yymin, float Yymax, float Expo)
906{

908 xx = pow(xx * xx, Expo);
