162// This function converts an int to a char[] array, then adds a comma, a dot, or nothing at the end.
163// It builds the char[] array at a pointer (*s) Where there MUST be enough space for all characters plus a zero terminator.
164// It dates for a very early time when I didn't know about standard library functions!
165// But it works just fine, so it says in.
166
167FASTRUN char* Str(char* s, int n, int comma) // comma = 0 for nothing, 1 for a comma, 2 for a dot.
168{
169 int r, i, m, flag;
170 char cma[] = ",";
171 char dot[] = ".";
172
173 flag = 0;
174 i = 0;
175 m = 1000000000;
176 if (n < 0) {
177 s[0] = '-';
178 i = 1;
179 n = -n;
180 }
181 if (n == 0) {
182 s[0] = 48;
183 s[1] = 0;
184
185 if (comma == 1) {
186 strcat(s, cma);
187 }
188 if (comma == 2) {
189 strcat(s, dot);
190 }
191 return s;
192 }
193 while (m >= 1) {
194 r = n / m;
195 if (r > 0) {
196 flag = 1;
197 } // first digit
198 if (flag == 1) {
199 s[i] = 48 + r;
200 ++i;
201 s[i] = 0;
202 }
203 n -= (r * m);
204 m /= 10;
205 }
206 if (comma == 1) {
207 strcat(s, cma);
208 }
209 if (comma == 2) {
210 strcat(s, dot);
211 }
212 return s;
213}

275void FixDeltaGMTSign()
276{
277 if (DeltaGMT < -24) DeltaGMT = 0; // Undefined value?f

332void AdjustDateTime(uint8_t MinChange, uint8_t HourChange, uint8_t YearChange, uint8_t MonthChange, uint8_t DateChange)
333{
334 ReadTheRTC();
335 Gminute += MinChange;
336 if (Gminute > 59) {
337 Gminute = 0;
338 if (Ghour < 23) {
339 ++Ghour;
340 }
341 }
342 if (Gminute < 1) {
343 Gminute = 0;
344 if (Ghour > 0) {
345 --Ghour;
346 }
347 }
348 Ghour += HourChange;
349 if (Ghour < 0) Ghour = 0;
350 if (Ghour > 23) Ghour = 23;
351 Gyear += YearChange;
352 if (Gyear < 0) Gyear = 0;
353 if (Gyear > 99) Gyear = 99;
354 Gmonth += MonthChange;
355 if (Gmonth < 1) Gmonth = 1;
356 if (Gmonth > 12) Gmonth = 12;
357 GmonthDay += DateChange;
358 if (GmonthDay < 1) GmonthDay = 1;
359 if (GmonthDay > 31) GmonthDay = 31;
360 SetTheRTC();
361}

505bool getDate(const char* str)
506{
507 char Month[12];
508 int Day, Year;
509 uint8_t monthIndex;
510 const char* monthName[12] = {
511 "Jan", "Feb", "Mar", "Apr", "May", "Jun",
512 "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
513 if (sscanf(str, "%s %d %d", Month, &Day, &Year) != 3) return false;
514 for (monthIndex = 0; monthIndex < 12; ++monthIndex) {
515 if (strcmp(Month, monthName[monthIndex]) == 0) break;
516 }
517 if (monthIndex >= 12) return false;
518 tm.Day = Day;
519 tm.Month = monthIndex + 1;
520 tm.Year = CalendarYrToTm(Year);
521 return true;
522}

536void ReadTime()
537{
538
539 static char month[12][15] = {"January", "February", "March", "April", "May", "June", "July", "August", "Sept", "October", "November", "December"};
540 static char ShortMonth[12][7] = {"Jan. ", "Feb. ", "Mar. ", "Apr. ", "May ", "June ", "July ", "Aug. ", "Sept ", "Oct. ", "Nov. ", "Dec. "};
541
542 char NB[10];
543 char TimeString[80];
544 char Space[] = " ";
545 char colon[] = ":";
546 char zero[] = "0";
547 char DateTime[] = "DateTime";
548
549 uint8_t DisplayedHour;
550 FixDeltaGMTSign();
551 if (CurrentView == FRONTVIEW || CurrentView == OPTIONVIEW2) {
552 if (RTC.read(tm)) {
553 strcpy(TimeString, Str(NB, tm.Day + DateFix, 0));
554 if (CurrentView == OPTIONVIEW2)
555 {
556 if ((tm.Day) < 10) {
557 strcat(TimeString, Space); // to align better the rest of the data
558 strcat(TimeString, Space);
559 }
560 }
561 strcat(TimeString, Space);
562 if (CurrentView == OPTIONVIEW2)
563 {
564 strcat(TimeString, ShortMonth[tm.Month - 1]);
565 }
566 else
567 {
568 strcat(TimeString, month[tm.Month - 1]);
569 }
570 strcat(TimeString, Space);
571 strcat(TimeString, (Str(NB, tmYearToCalendar(tm.Year), 0)));
572 strcat(TimeString, Space);
573 DisplayedHour = tm.Hour + DeltaGMT;
574 DateFix = 0;
575 if (DisplayedHour > 24) {
576 DisplayedHour -= 24;
577 DateFix = 1;
578 }
579 if (DisplayedHour < 0) {
580 DisplayedHour += 24;
581 DateFix = -1;
582 }
583 if (MayBeAddZero(DisplayedHour)) strcat(TimeString, zero);
584 strcat(TimeString, Str(NB, DisplayedHour, 0));
585 strcat(TimeString, colon);
586 if (MayBeAddZero(tm.Minute)) strcat(TimeString, zero);
587 strcat(TimeString, Str(NB, tm.Minute, 0));
588 strcat(TimeString, colon);
589 if (MayBeAddZero(tm.Second)) strcat(TimeString, zero);
590 strcat(TimeString, Str(NB, tm.Second, 0));
591 SendText(DateTime, TimeString);
592 }
593 }

951/*********************************************************************************************************************************/
952uint16_t GetSuccessRate()
953{
954 uint16_t Total = 0;
955 uint16_t SuccessRate;
956 uint16_t Perfection = (PERFECTPACKETSPERSECOND * (uint16_t)ConnectionAssessSeconds);
957
958 for (uint16_t i = 0; i < Perfection; ++i) { // PERFECTPACKETSPERSECOND packets per second are either good or bad
959 Total += PacketsHistoryBuffer[i];
960 }
961
962 Total += ((Perfection - Total) / 2); // about half made it but were simply unacknowledged
963 SuccessRate = (Total * 100) / Perfection; // return a percentage of total good packets
964 return SuccessRate;
965}
