11FASTRUN void CreateTimeStamp(char* DateAndTime)
12{
13 char NB[10];
14 char zero[] = "0";
15 char Colon[] = "."; // a dot!
16 char Slash[] = "/";
17 char null[] = "";
18 char space[] = " ";
19
20 if (RTC.read(tm)) {
21 strcpy(DateAndTime, null);
22
23 if (MayBeAddZero(tm.Day)) strcat(DateAndTime, zero);
24 strcat(DateAndTime, Str(NB, tm.Day, 0));
25 strcat(DateAndTime, Slash);
26 if (MayBeAddZero(tm.Month)) strcat(DateAndTime, zero);
27 strcat(DateAndTime, Str(NB, tm.Month, 0));
28 strcat(DateAndTime, Slash);
29 strcat(DateAndTime, (Str(NB, tmYearToCalendar(tm.Year), 0)));
30 strcat(DateAndTime, space);
31
32 if (MayBeAddZero(tm.Hour)) strcat(DateAndTime, zero);
33 strcat(DateAndTime, Str(NB, tm.Hour, 0));
34 strcat(DateAndTime, Colon);
35 if (MayBeAddZero(tm.Minute)) strcat(DateAndTime, zero);
36 strcat(DateAndTime, Str(NB, tm.Minute, 0));
37 strcat(DateAndTime, Colon);
38 if (MayBeAddZero(tm.Second)) strcat(DateAndTime, zero);
39 strcat(DateAndTime, Str(NB, tm.Second, 0));
40 }
41}
