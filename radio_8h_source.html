230void Reconnect(){
231 uint32_t SearchStartTime = 0;

233 FailSafeSent = false;
234 CurrentRadio->stopListening();
235 delay(1);

245 ThisRadio = 1;
246 } else {
247 CurrentRadio = &Radio2;
248 ThisRadio = 2;
249 }
250 ProdRadio();
251#endif // defined (SECOND_TRANSCEIVER)
252 ListenALittle(LISTEN_PERIOD);
253 if (!Connected) {
254 if ((millis() - SearchStartTime) > FAILSAFE_TIMEOUT){
255 if (!FailSafeSent){
256 FailSafe();
257 FailSafeSent = true; // Once is enough
258 }
259 }
260 }
261 }
262 ReconnectedMoment = millis(); // Save this moment, then don't move a servo for a few ms ...
263}
264/************************************************************************************************************/
265// This function checks the time since last hop.
266// If it's time to HOP, it sets the high bit in AckPayload.Purpose and both ends then HOP to new channel before next packet.
267// The other 7 BITS of AckPayload.Purpose still dictate the Payload's function ( < 127 possibities. )
268// This now happens for *every* AckPayload, which can and does return telemetry data as well as this hoptime information.
269// Hence a single BIT now directs the transmitter to hop.
270
271void CheckIfItsHopTime(){
272 AckPayload.Purpose &= 0x7f; // Clear the HOP flag
273 if ((millis() - HopStart) >= HOPTIME){ // Time to hop??
274 AckPayload.Purpose |= 0x80; // Yes. So set the HOP flag leaving lower 7 bits unchanged
275 ++NextChannelNumber; // Move up the channels' array
276 if (NextChannelNumber >= FREQUENCYSCOUNT) NextChannelNumber = 1; // If needed, wrap the channels' array pointer
277 AckPayload.Byte5 = NextChannelNumber; // Tell the transmitter which element of the array to use next.
278 GetNextFrequency(); // Get the actual channel number from the array.
279 HopNow = true; // Set local flag and hop when ready BUT NOT BEFORE.
280 }
281}

307void LoadAckPayload()
308{
309 uint8_t MaxAckP = 0; // 0 if only RX
310 AckPayload.Purpose &= 0x7F; // NOTE: The HIGH BIT of "purpose" bit is the HOPNOW flag. It gets set only when it's time to hop.
311 ++AckPayload.Purpose;
312 if (INA219_CONNECTED) MaxAckP = 1;
313 if (SENSOR_HUB_CONNECTED) MaxAckP = 14; // its 14 + GPS
314 if (AckPayload.Purpose > MaxAckP) AckPayload.Purpose = 0; // wrap after max
315 switch (AckPayload.Purpose) {
316 case 0:
317 LoadVersioNumber();
318 break;
319 case 1:
320 SendToAckPayload(INA219Volts);
321 break;
322 case 2:
323 SendToAckPayload(BaroAltitude);
324 break;
325 case 3:
326 SendToAckPayload(BaroTemperature);
327 break;
328 case 4:
329 SendToAckPayload (LatitudeGPS);
330 break;
331 case 5:
332 SendToAckPayload (LongitudeGPS);
333 break;
334 case 6:
335 SendToAckPayload (AngleGPS);
336 break;
337 case 7:
338 SendToAckPayload(SpeedGPS);
339 break;
340 case 8:
341 SendToAckPayload(GpsFix);
342 break;
343 case 9:
344 SendToAckPayload(AltitudeGPS);
345 break;
346 case 10:
347 SendToAckPayload(DistanceGPS);
348 break;
349 case 11:
350 SendToAckPayload(CourseToGPS);
351 break;

366/*
367 * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
368 * @param uncompressed_buf[in]
369 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
370 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
371 */
372void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, int uncompressed_size)
373{
374 int p = 0;
375 for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
376 uncompressed_buf[p] = compressed_buf[l] >> 4;
377 ++p;
378 uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
379 ++p;
380 uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
381 ++p;
382 uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
383 ++p;
384 }
385}
