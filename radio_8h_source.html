99/*
100 * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
101 * @param uncompressed_buf[in]
102 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
103 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
104 */
105void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, uint8_t uncompressed_size)
106{
107 uint8_t p = 0;
108 for (uint8_t l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
109 uncompressed_buf[p] = compressed_buf[l] >> 4;
110 ++p;
111 uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
112 ++p;
113 uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
114 ++p;
115 uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
116 ++p;
117 }
118}

180bool ReadData()
181{
182 uint8_t MAXSMALLACKS = 100;
183 static uint8_t AckCounter = 0;
184 Connected = false;
185 if (CurrentRadio->available(&Pipnum))
186 {
187 uint8_t DynamicPayloadSize = CurrentRadio->getDynamicPayloadSize(); // Get the size of the new data (14)
188 CurrentRadio->flush_tx(); // This avoids a lockup that happens when the FIFO gets full
189
190 if (DynamicPayloadSize == 16) { // If size = 16 it must be a parameter packet if over 10 bytes // todo: Send ACK that says "I got a parameter" heer!!
191 // Look1(millis());
192 // Look1(" ");
193 // Look1("DynamicPayloadSize: ");
194 // Look(DynamicPayloadSize);
195 AckCounter = 0; // force small ack
196 }
197 if (AckCounter > MAXSMALLACKS) { // heer
198 AckCounter = 0;
199 LoadAckPayload();
200 CurrentRadio->writeAckPayload(1, &AckPayload, AckPayloadSize); // send big PAYLOAD EVERY 100th time (2 per second)
201 }else{
202 LoadShortAckPayload();
203 CurrentRadio->writeAckPayload(1, &AckPayload, 2); // send VERY little 99% of the time
204 ++AckCounter;
205 }
206 DelayMillis(1); // 1 ms delay
207 CurrentRadio->read(&DataReceived, DynamicPayloadSize); // ** >> Read new data from master << ** // Get the size of the new data (14)
208 Connected = true;
209 NewData = true;
210 if (Connected) UseReceivedData(DynamicPayloadSize);
211 }
212 return Connected;
213}

296/************************************************************************************************************/
297// This function compares the just-received pipe with several of the previous ones
298// if it matches most of them then its probably not corrupted.
299
300bool ValidateNewPipe()
301{
302
303 uint8_t MatchedCounter = 0;
304
305 if (pcount < 2) return false; // ignore first few
306
307 PreviousNewPipes[PreviousNewPipesIndex] = NewPipeMaybe;
308 PreviousNewPipesIndex++;
309 if (PreviousNewPipesIndex > PIPES_TO_COMPARE) PreviousNewPipesIndex = 0;
310
311 for (int i = 0; i < PIPES_TO_COMPARE; ++i) {
312 if (NewPipeMaybe == PreviousNewPipes[i]) ++MatchedCounter;
313 }
314
315 if (MatchedCounter >= 2) return true;
316 return false;
317}

559 uint8_t PreviousRadio = ThisRadio;

564
565 while (!Connected) {

578 CurrentRadio->setChannel(ReconnectChannel);
579 CurrentRadio->startListening();

582 if (Attempts < MAXTRIESPERTRANSCEIVER) {

584 }

586

648/************************************************************************************************************/
649// This function checks the time since last hop.
650// If it's time to HOP, it sets the high bit in AckPayload.Purpose and both ends then HOP to new channel before next packet.
651// The other 7 BITS of AckPayload.Purpose dictate the Payload's function (therefore 127 possibities.)
652// This happens for *every* AckPayload, which return telemetry data as well as this hoptime information.
653// Hence a single BIT directs the transmitter to hop.
654
655void CheckWhetherItsTimeToHop()
656{
657 AckPayload.Purpose &= 0x7f; // Clear the HOP flag
658 if ((millis() - HopStart) >= HOPTIME) { // Time to hop??
659 AckPayload.Purpose |= 0x80; // Yes. So set the HOP flag leaving lower 7 bits unchanged
660 IncChannelNumber();
661 HopNow = true; // Set local flag and hop when ready BUT NOT BEFORE.
662 }

712/************************************************************************************************************/
713// The unique Mac address of this Teensy 4.0 is sent to transmitter while binding to identify this model.
714// This is to avoid the wrong model memory being used.
715
716void SendMacAddress()
717{
718 union
719 {
720 uint32_t Val32[2];
721 uint8_t Val8[8]; // the highest two bytes will always be zero. We didn't need all 8.
722 } ThisUnion;
723 uint8_t MaxAckP = 1; // only packets 0 and 1 are needed here.
724
725 ++MacAddressSentCounter;
726
727 AckPayload.Purpose &= 0x7F;
728 ++AckPayload.Purpose;
729 if (AckPayload.Purpose > MaxAckP) AckPayload.Purpose = 0; // wrap after max
730
731 for (int i = 0; i < 8; ++i) ThisUnion.Val8[i] = MacAddress[i];
732
733 switch (AckPayload.Purpose) {
734 case 0:
735 SendIntToAckPayload(ThisUnion.Val32[0]);
736 break;
737 case 1:
738 SendIntToAckPayload(ThisUnion.Val32[1]);
739 break;
740
