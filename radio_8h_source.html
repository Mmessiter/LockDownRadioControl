234void Reconnect()
235{
236 uint32_t TryTimer;
237 uint8_t ReconnectAttempts = 0;
238
239 SearchStartTime = millis();
240 FailSafeSent = false;
241 CurrentRadio->stopListening();
242 delay(1);
243 CurrentRadio->setChannel(RECONNECT_CH);
244 CurrentRadio->startListening();
245 delay(1);
246 if (CurrentRadio->available())
247 {
248 Connected = true;
249 }
250 while (!Connected)
251 {
252 ++ReconnectAttempts;
253#ifdef SECOND_TRANSCEIVER
254 if (ReconnectAttempts > 2)
255 {
256 CurrentRadio->stopListening();
257 delay(1);
258 if (ThisRadio == 2)
259 {
260 CurrentRadio = &Radio1;
261 ThisRadio = 1;
262 }
263 else
264 {
265 CurrentRadio = &Radio2;
266 ThisRadio = 2;
267 }
268 ProdRadio();
269 ReconnectAttempts = 0;
270 }
271#endif // defined (SECOND_TRANSCEIVER)
272 TryTimer = millis();
273 while ((!CurrentRadio->available()) && (millis() - TryTimer) < 100) {
274 } // wait a mo during attempt to connect ...
275 if (CurrentRadio->available()) Connected = true;
276 if (!Connected)
277 {
278 StillSearchingTime = millis() - SearchStartTime;
279 if (StillSearchingTime > FAILSAFE_TIMEOUT)
280 {
281 if (!FailSafeSent)
282 {
283 FailSafe();
284 FailSafeSent = true; // Once is enough
285 }
286 }
287 }
288 }
289 ConnectionStart = millis();
290 StillSearchingTime = 0;
291 ReconnectedMoment = ConnectionStart; // Save this moment, then don't move a servo for a few ms ...
292}

391/*
392 * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
393 * @param uncompressed_buf[in]
394 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
395 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
396 */
397void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, int uncompressed_size)
398{
399 int p = 0;
400 for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
401 uncompressed_buf[p] = compressed_buf[l] >> 4;
402 ++p;
403 uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
404 ++p;
405 uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
406 ++p;
407 uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
408 ++p;
409 }
