365 void LoadAckPayload()
366 {
367  AckPayload.Purpose &= 0x7F; // Clear hi bit ( = do not ignore)
368  ++AckPayload.Purpose; // 0 = ...
369  // 1 = Version number
370  // 2 = Time stamp for FHSS
371  // 3 = RX lipo Voltage
372  // More later ... CAN EXPAND TO 127 if needed :-)!
373 
374  if (AckPayload.Purpose > 7) AckPayload.Purpose = 0; // 7 is currently the max
375 
376  switch (AckPayload.Purpose){
377  case 0: // if 0 send synch time stamp
378  LoadTimeStamp();
379  break;
380  case 1: // if 1 send version info AND RX number
381  LoadVersioNumber();
382  break;
383  case 2: // if 2 send synch time stamp
384  LoadTimeStamp();
385  break;
386  case 3: // if 3 send send rx lipo volts
387  LoadRXVolts();
388  break;
389  case 4: // if 4 send synch time stamp
390  LoadTimeStamp();
391  break;
392  case 5: // if 5 send synch time stamp
393  LoadAltitude();
394  break;
