570 /*
571  * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
572  * @param uncompressed_buf[in]
573  * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
574  * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
575  */
576 void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, int uncompressed_size)
577 {
578  int p = 0;
579  for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
580  uncompressed_buf[p] = compressed_buf[l] >> 4;
581  ++p;
582  uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
583  ++p;
584  uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
585  ++p;
586  uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
587  ++p;
588  }
589 }
