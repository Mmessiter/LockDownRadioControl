267void Reconnect(){ // This is called when contact is lost, to reconnect ASAP
268 uint32_t SearchStartTime = millis();;
269 uint8_t ReconnectChannel = * (FHSSChPointer + ReconnectIndex); // Get a reconnect channel
270 uint8_t PreviousRadio = ThisRadio;
271#ifdef SECOND_TRANSCEIVER
272 uint8_t Attempts = 0;
273#endif
274 if (ThisRadio == 1) RX1TotalTime += (millis() - ReconnectedMoment); // keep track of how long on each
275 if (ThisRadio == 2) RX2TotalTime += (millis() - ReconnectedMoment);
276 while (!Connected) {
277 if (BoundFlag) KeepSbusHappy(); // Some SBUS systems timeout FAST, so resend old data to keep it happy
278 CurrentRadio->stopListening();
279 delayMicroseconds(500); // NEEDED!
280 ReconnectChannel = * (FHSSChPointer + ReconnectIndex); // Get a reconnect channel
281 ++ ReconnectIndex;
282 if (ReconnectIndex >= RECONNECT_CHANNELS_COUNT + RECONNECT_CHANNELS_START) ReconnectIndex = RECONNECT_CHANNELS_START;
283 CurrentRadio->setChannel(ReconnectChannel);
284 TryToConnectNow();
285
286#ifdef SECOND_TRANSCEIVER
287 ++ Attempts;
288 if (Attempts >= 2){if (!Connected) TryTheOtherTransceiver(ReconnectChannel); Attempts = 0;}
289#endif
290 if (!Connected) {
291 if ((millis() - SearchStartTime) > FAILSAFE_TIMEOUT){
292 if (!FailSafeSent) FailSafe();
293 }
294 }
295 }
296 FailSafeSent = false;
297 if (PreviousRadio != ThisRadio) ++ RadioSwaps; // Count the radio swaps
298 ReconnectedMoment = millis(); // Save this moment, then don't move a servo for 20 ms ...
299#ifdef DB_RXTIMERS
300 Serial.print ("Transceiver1 use so far: ");
301 Serial.print (RX1TotalTime/1000);
302 Serial.println (" seconds");
303 Serial.print ("Transceiver2 use so far: ");
304 Serial.print (RX2TotalTime/1000);
305 Serial.println (" seconds");
306 Serial.print ("Now connected on transceiver number: ");
307 Serial.print (ThisRadio);
308 Serial.println (" ...");
309 Serial.println ("");
310#endif

313/************************************************************************************************************/
314// This function checks the time since last hop.
315// If it's time to HOP, it sets the high bit in AckPayload.Purpose and both ends then HOP to new channel before next packet.
316// The other 7 BITS of AckPayload.Purpose dictate the Payload's function (therefore 127 possibities.)
317// This happens for *every* AckPayload, which return telemetry data as well as this hoptime information.
318// Hence a single BIT now directs the transmitter to hop.
319
320void CheckWhetherItsTimeToHop(){
321 AckPayload.Purpose &= 0x7f; // Clear the HOP flag
322 if ((millis() - HopStart) >= HOPTIME){ // Time to hop??
323 AckPayload.Purpose |= 0x80; // Yes. So set the HOP flag leaving lower 7 bits unchanged
324 ++NextChannelNumber; // Move up the channels' array
325 if (NextChannelNumber >= FrequencyCount) NextChannelNumber = 1; // If needed, wrap the channels' array pointer
326 AckPayload.Byte5 = NextChannelNumber; // Tell the transmitter which element of the array to use next.
327 NextChannel = * (FHSSChPointer + NextChannelNumber); // Get the actual channel number from the array.
328 HopNow = true; // Set local flag and hop when ready BUT NOT BEFORE.
329 }
330}

356void LoadAckPayload()
357{
358 uint8_t MaxAckP = 4; // 4 if only RX
359 AckPayload.Purpose &= 0x7F; // NOTE: The HIGH BIT of "purpose" bit is the HOPNOW flag. It gets set only when it's time to hop.
360 ++AckPayload.Purpose;
361 if (INA219_CONNECTED) MaxAckP = 5;
362 if (SENSOR_HUB_CONNECTED) MaxAckP = 18; // its 14 + GPS
363 if (AckPayload.Purpose > MaxAckP) AckPayload.Purpose = 0; // wrap after max
364 switch (AckPayload.Purpose) {
365 case 0:
366 SendVersionNumberToAckPayload();
367 break;
368 case 1:
369 SendToAckPayload (SbusRepeats);
370 break;
371 case 2:
372 SendToAckPayload (RadioSwaps);

427/*
428 * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
429 * @param uncompressed_buf[in]
430 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
431 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
432 */
433void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, uint8_t uncompressed_size)
434{
435 uint8_t p = 0;
436 for (uint8_t l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
437 uncompressed_buf[p] = compressed_buf[l] >> 4;
438 ++p;
439 uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
440 ++p;
441 uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
442 ++p;
443 uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
444 ++p;
445 }
446}
