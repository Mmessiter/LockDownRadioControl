445 /*
446  * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
447  * @param uncompressed_buf[in]
448  * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
449  * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
450  */
451 void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, int uncompressed_size)
452 {
453  int p = 0;
454  for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
455  uncompressed_buf[p] = compressed_buf[l] >> 4;
456  ++p;
457  uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
458  ++p;
459  uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
460  ++p;
461  uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
462  ++p;
463  }
464 }
