291FASTRUN void Reconnect()
292{ // This is called when contact is lost, to reconnect ASAP
293 uint32_t SearchStartTime = millis();
294 uint8_t ReconnectChannel = *(FHSSRecoveryPointer + ReconnectIndex); // Get a reconnect channel
295 uint8_t PreviousRadio = ThisRadio;
296 uint8_t Attempts = 0;
297
298 if (ThisRadio == 1) RX1TotalTime += (millis() - ReconnectedMoment); // keep track of how long on each
299 if (ThisRadio == 2) RX2TotalTime += (millis() - ReconnectedMoment);
300
301 while (!Connected) {
302 if (BoundFlag) KeepSbusHappy(); // Some SBUS systems timeout FAST, so resend old data to keep it happy
303 CurrentRadio->stopListening();
304 CurrentRadio->flush_tx();
305 CurrentRadio->flush_rx();
306
307 delayMicroseconds(1000); // NEEDED!
308
309 ReconnectChannel = *(FHSSRecoveryPointer + ReconnectIndex); // Get a reconnect channel
310 ++ReconnectIndex;
311 if (ReconnectIndex >= RECONNECT_CHANNELS_COUNT + RECONNECT_CHANNELS_START) ReconnectIndex = RECONNECT_CHANNELS_START;
312 CurrentRadio->setChannel(ReconnectChannel);
313 ++Attempts;
314 if (Attempts < 3) TryToConnectNow();
315 if (!Connected) {
316
317#ifdef SECOND_TRANSCEIVER
318 if (Attempts >= 3) {
319 TryTheOtherTransceiver(ReconnectChannel);
320 Attempts = 0;
321 }
322#else
323 if (Attempts >= 3) {
324 ProdRadio(ReconnectChannel); // This avoids a lockup of the nRF24L01+ !
325 Attempts = 0;
326 }
327#endif
328 if ((millis() - SearchStartTime) > FAILSAFE_TIMEOUT) {
329 if (!FailSafeSent) FailSafe();
330 }
331 }
332 } // cannot pass here if not connected
333
334 // must have connected by here
335
336 FailSafeSent = false;
337 if (PreviousRadio != ThisRadio) ++RadioSwaps; // Count the radio swaps
338 ReconnectedMoment = millis(); // Save this moment
339 if (FailedSafe){
340 FailedSafe = false;
341 NewConnectionMoment = millis();
342 }
343#ifdef DB_RXTIMERS
344 Serial.print("Transceiver1 use so far: ");
345 Serial.print(RX1TotalTime / 1000);
346 Serial.println(" seconds");
347 Serial.print("Transceiver2 use so far: ");
348 Serial.print(RX2TotalTime / 1000);
349 Serial.println(" seconds");
350 Serial.print("Now connected on transceiver number: ");
351 Serial.print(ThisRadio);
352 Serial.println(" ...");
353 Serial.println("");
354#endif
355}
356/************************************************************************************************************/
357// This function checks the time since last hop.
358// If it's time to HOP, it sets the high bit in AckPayload.Purpose and both ends then HOP to new channel before next packet.
359// The other 7 BITS of AckPayload.Purpose dictate the Payload's function (therefore 127 possibities.)
360// This happens for *every* AckPayload, which return telemetry data as well as this hoptime information.
361// Hence a single BIT directs the transmitter to hop.
362
363void CheckWhetherItsTimeToHop()
364{
365 AckPayload.Purpose &= 0x7f; // Clear the HOP flag
366 if ((millis() - HopStart) >= HOPTIME) { // Time to hop??
367 AckPayload.Purpose |= 0x80; // Yes. So set the HOP flag leaving lower 7 bits unchanged
368 ++NextChannelNumber; // Move up the channels' array
369 if (NextChannelNumber >= FrequencyCount) NextChannelNumber = 1; // If needed, wrap the channels' array pointer
370 AckPayload.Byte5 = NextChannelNumber; // Tell the transmitter which element of the array to use next.
371 NextChannel = *(FHSSChPointer + NextChannelNumber); // Get the actual channel number from the array.
372 HopNow = true; // Set local flag and hop when ready BUT NOT BEFORE.
373 }
374}

419// The unique Mac address of this Teensy 4.0 is sent to transmitter while binding to identify this model.
420// This is to avoid the wrong model memory being used.
421
422void SendMacAddress()
423{
424 union {
425 uint32_t Val32[2];
426 uint8_t Val8[8]; // the highest two bytes will always be zero. We didn't need all 8.
427 } ThisUnion;
428 uint8_t MaxAckP = 1; // only packets 0 and 1 are needed here.
429
430 AckPayload.Purpose &= 0x7F;
431 ++AckPayload.Purpose;
432 if (AckPayload.Purpose > MaxAckP) AckPayload.Purpose = 0; // wrap after max
433
434 for (int i = 0; i < 8; ++i) ThisUnion.Val8[i] = MacAddress[i];
435
436 switch (AckPayload.Purpose) {
437 case 0:
438 SendIntToAckPayload(ThisUnion.Val32[0]);
439 break;
440 case 1:
441 SendIntToAckPayload(ThisUnion.Val32[1]);
442 break;
443 default:
444 break;
445 }
446}

525/************************************************************************************************************/
526/*
527 * Decompresses uint16_t* buffer values (each with 12 bit resolution - the lower 12 bits).
528 * @param uncompressed_buf[in]
529 * @param compressed_buf[out] Must have allocated 3/4 the size of uncompressed_buf
530 * @param uncompressed_size Size is in units of uint16_t (aka word or unsigned short)
531 */
532void Decompress(uint16_t* uncompressed_buf, uint16_t* compressed_buf, uint8_t uncompressed_size)
533{
534 uint8_t p = 0;
535 for (uint8_t l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
536 uncompressed_buf[p] = compressed_buf[l] >> 4;
537 ++p;
538 uncompressed_buf[p] = (compressed_buf[l] & 0xf) << 8 | compressed_buf[l + 1] >> 8;
539 ++p;
540 uncompressed_buf[p] = (compressed_buf[l + 1] & 0xff) << 4 | compressed_buf[l + 2] >> 12;
541 ++p;
542 uncompressed_buf[p] = compressed_buf[l + 2] & 0xfff;
543 ++p;
544 }
