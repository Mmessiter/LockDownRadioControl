75void GetMasterAck() // Here Pupil gets Ack from master while Pupil is in control
76// if ack is 'S' then master is taking control back
77// if ack is 'O' then master is ok with pupil staying in control
78{
79 char AckSpecial[2] = " "; // simple ack
80 Radio1.read(&AckSpecial, 2);
81 if (AckSpecial[0] == 'O') return; // OK to continue sending. no action needed
82 if (AckSpecial[0] == 'S') { // PUPIL -> LISTEN HEER <<<<<< *******
83 PlaySound(MASTERMSG);
84 StartBuddyListen(0); // 'S' from Master = Shut Up now, send nothing
85 DelayWithDog(LONGER_DELAY);
86 }
87 FlushFifos();
88 MasterDetected(true); // Master is alive
89 return;
90}

105void SendSpecialPacket(bool IamMaster) // here the sender sends to other tx while in control. This is called about 5 times a second.
106// hence the sender always stays in send mode, and the silent tx stays in receive (LISTEN) mode, only switching to send when takes control.
107{
108 uint8_t Master_in_Control[2] = "S"; // = Shut Up, send nothing
109 uint8_t Pupil_in_Control[2] = "O"; // = OK to send data
110 uint8_t Pupil_is_Alive[2] = "P";
111 FlushFifos();
112 Radio1.setChannel(SPECIAL_PACKET_CHANNEL);
113 delayMicroseconds(SHORT_DELAY);
114 Radio1.stopListening();
115 delayMicroseconds(SHORT_DELAY);
116
117 if (IamMaster) {
118 if (!BuddyON)
119 {
120 delayMicroseconds(SHORT_DELAY);
121 if (Radio1.write(&Master_in_Control, sizeof(Master_in_Control))) {
122 GetPupilAck(); // get ack from pupil
123 }
124 else {
125 PupilDetected(false);
126 }
127 }
128 if (BuddyON) // PUPIL goes into CONTROL
129 {
130 if (Radio1.write(&Pupil_in_Control, sizeof(Pupil_in_Control))) { // SEND O to Pupil. If we can't, he's dead
131 GetPupilAck();
132 StartBuddyListen(1); // MASTER -> LISTEN (Pupil took control) HEER <<<<<< *******
133 PlaySound(BUDDYMSG);
134 DelayWithDog(LONGER_DELAY);
135 LastPassivePacketTime = millis();
136 return;
137 }
138 else {
139 PupilDetected(false);
140 }
141 }
142 }
143 if (!IamMaster) { // pupil area when in control *************************************************************
144 if (Radio1.write(&Pupil_is_Alive, sizeof(Pupil_is_Alive))) { // send P to master
145 MasterDetected(true);
146 delayMicroseconds(SHORT_DELAY);
147 GetMasterAck();
148 }
149 else { // master must be either back in control, or ** dead ***. Pupil will shut up in either case
150 StartBuddyListen(0); // <<<<<<<<<<<<<<<<<<< PUPIL -> LISTEN <<<<<< *******
151 MasterDetected(false);
152 PlaySound(MASTERMSG);
153 DelayWithDog(LONGER_DELAY);
154 FlushFifos();
155 }
156 if (CurrentMode == LISTENMODE) return; // control might have ceased
157 }
158 Radio1.setChannel(CurrentChannel); // restore the proper frequency
159 delayMicroseconds(SHORT_DELAY);
160 Radio1.stopListening();
161 delayMicroseconds(SHORT_DELAY);
162}

166void GetSpecialPacket(bool IamMaster) // here the passive tx gets from active tx
167{
168 char DataPacket[2] = " ";
169 char Master_in_Control[2] = "S"; // = Shut Up, send nothing
170 char Pupil_in_Control[2] = "O"; // = OK to send data
171 char Ack[2] = "P"; // Pupil ack
172 bool TakeControlBackNow = false;
173
174 if (Radio1.available()) {
175 if (IamMaster) { // master here
176 if (BuddyON)
177 {
178 Ack[0] = 'O'; // ok to continue sending - you're in charge
179 TakeControlBackNow = false;
180 }
181 if (!BuddyON)
182 {
183 Ack[0] = 'S'; // shut up now as buddy is now off.... *********
184 TakeControlBackNow = true;
185 }
186 }
187 delayMicroseconds(SHORT_DELAY);
188 Radio1.writeAckPayload(1, &Ack, 2); // Acknowledge the packet
189 delayMicroseconds(SHORT_DELAY);
190 Radio1.read(&DataPacket, sizeof(DataPacket));
191 if (!IamMaster) { // pupil here
192 MasterDetected(true);
193 if (DataPacket[0] == Master_in_Control[0]) {
194 } // no action needed
195 if (DataPacket[0] == Pupil_in_Control[0]) {
196 StopBuddyListen(0); // PUPIL -> CONTROL HEER <<<<<< *******
197 DelayWithDog(LONGER_DELAY);
198 return;
199 }
200 }
201 DelayWithDog(LONGER_DELAY);
202 if (TakeControlBackNow) {
203 DelayWithDog(LONGER_DELAY);
204 StopBuddyListen(1); // MASTER RECLAIMS CONTROL HEER <<<<<< *******
205 }
206 LastPassivePacketTime = millis();
207 FlushFifos();
208 }
209 else { // no packet arrived Pupil's dead!!
210 if (IamMaster) {
211 if (millis() - LastPassivePacketTime > INTERBUDDYRATE + 50) {
212 StopBuddyListen(1); // MASTER RECLAIMS CONTROL HEER <<<<<< *******
213 }
214 }
215 if (!IamMaster) {
216 MasterDetected(false);
217 }
218 }
219}
