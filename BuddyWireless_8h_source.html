18// This function is called by Pupil and the Master was Detected - or not Detected.
19// After 6 failed detections, the Master is declared dead, and a message is changed on front view.
20void MasterDetected(bool Detected)
21{
22 static uint16_t LostMasterCount = 0;
23 char Mlost[] = "Master not found";
24 char Mfound[] = "Master found!";
25 char wb[] = "wb"; // wb is the name of the label on front view
26 char YesVisible[] = "vis wb,1";
27
28 if (Detected) {
29 LostMasterCount = 0;
30 LastPassivePacketTime = millis(); // reset the timer
31 if (MasterIsAlive != 1) { // MasterIsAlive = 1 means Master was already alive
32 SendText(wb, Mfound);
33 SendCommand(YesVisible);
34 MasterIsAlive = 1;
35 }
36 }
37 else {
38 ++LostMasterCount;
39 if (LostMasterCount > LOSTCONTACTTHRESHOLD) {
40 if (MasterIsAlive != 2) { // MasterIsAlive = 2 means Master was already dead
41 SendText(wb, Mlost);
42 SendCommand(YesVisible);
43 MasterIsAlive = 2;
44 }
45 }
46 }
47}

82void GetMasterAck() // Here Pupil gets Ack from master while Pupil is in control
83// if ack is 'S' then master is taking control back
84// if ack is 'O' then master is ok with pupil staying in control
85{
86 char AckSpecial[2] = " "; // simple ack
87 Radio1.read(&AckSpecial, 2);
88 if (AckSpecial[0] == 'O') return; // OK to continue sending. no action needed
89 if (AckSpecial[0] == 'S') { // PUPIL -> LISTEN <<<<<< *******
90 StartBuddyListen(0); // 'S' from Master = Shut Up now, send nothing
91 PlaySound(MASTERMSG);
92 DelayWithDog(LONGER_DELAY);
93 }
94 MasterDetected(true); // Master is alive
95}

108void SendSpecialPacket(bool IamMaster) // here the sender sends to other tx while in control. This is called about 5 times a second.
109// hence the sender always stays in send mode, and the silent tx stays in receive (LISTEN) mode, only switching to send when takes control.
110{
111 uint8_t Master_in_Control[2] = "S"; // = Shut Up, send nothing
112 uint8_t Pupil_in_Control[2] = "O"; // = OK to send data
113 uint8_t Pupil_is_Alive[2] = "P";
114 FlushFifos();
115
116 if (IamMaster) Radio1.openWritingPipe(TeensyMACAddPipe ^ ENCRYPT_KEY); // send to encrypted pipe address
117 if (!IamMaster) Radio1.openWritingPipe(BuddyMACAddPipe ^ ENCRYPT_KEY); // send to encrypted pipe address
118 delayMicroseconds(SHORT_DELAY);
119
120 Radio1.setChannel(SPECIAL_PACKET_CHANNEL);
121 delayMicroseconds(SHORT_DELAY);
122 Radio1.stopListening();
123 delayMicroseconds(SHORT_DELAY);
124 if (IamMaster) {
125 if (!BuddyON)
126 {
127 delayMicroseconds(SHORT_DELAY);
128 if (Radio1.write(&Master_in_Control, sizeof(Master_in_Control))) {
129 GetPupilAck(); // get ack from pupil
130 }
131 else {
132 PupilDetected(false);
133 }
134 }
135 if (BuddyON) // PUPIL goes into CONTROL
136 {
137 if (Radio1.write(&Pupil_in_Control, sizeof(Pupil_in_Control))) { // SEND O to Pupil. If we can't, he's dead
138 GetPupilAck();
139 StartBuddyListen(1); // MASTER -> LISTEN (Pupil took control) <<<<<< *******
140 PlaySound(BUDDYMSG);
141 LastPassivePacketTime = millis();
142 return;
143 }
144 else {
145 PupilDetected(false);
146 }
147 }
148 }
149 if (!IamMaster) { // pupil area when in control *************************************************************
150 delayMicroseconds(SHORT_DELAY);
151 if (Radio1.write(&Pupil_is_Alive, sizeof(Pupil_is_Alive))) { // send P to master
152 GetMasterAck();
153 }
154 else { // master must be either back in control, or ** dead ***. Pupil will shut up in either case
155 StartBuddyListen(0); // <<<<<<<<<<<<<<<<<<< PUPIL -> LISTEN <<<<<< ******* THIS SHOULD NEVER OCCUR ?! will sort out later ... meanwhile it just works
156 MasterDetected(false);
157 PlaySound(MASTERMSG);
158 DelayWithDog(LONGER_DELAY);
159 FlushFifos();
160 }
161 if (CurrentMode == LISTENMODE) return; // control might have ceased
162 }
163
164 if (IamMaster) Radio1.openWritingPipe(TeensyMACAddPipe); // restore the proper pipe address
165 if (!IamMaster) Radio1.openWritingPipe(BuddyMACAddPipe); // restore the proper pipe address
166 delayMicroseconds(SHORT_DELAY);
167 Radio1.stopListening();
168 delayMicroseconds(SHORT_DELAY);
169 Radio1.setChannel(CurrentChannel); // restore the proper frequency
170 delayMicroseconds(SHORT_DELAY);
171 Radio1.stopListening();
172}

176void GetSpecialPacket(bool IamMaster) // here the passive tx gets from active tx. This function is called from main loop
177{
178 char DataPacket[2] = " ";
179 char Pupil_in_Control[2] = "O"; // = OK to send data
180 char Ack[2] = "P"; // Pupil ack
181 bool TakeControlBackNow = false;
182
183 if (Radio1.available()) {
184 if (IamMaster) { // master here
185 if (BuddyON)
186 {
187 Ack[0] = 'O'; // ok to continue sending - you're in charge
188 TakeControlBackNow = false;
189 }
190 if (!BuddyON)
191 {
192 Ack[0] = 'S'; // shut up NOW please, I'm taking over *********
193 TakeControlBackNow = true;
194 }
195 }
196 Radio1.writeAckPayload(1, &Ack, 2); // Acknowledge the packet
197 DelayWithDog(DELAYAFTERACK); // <- ** MUST ** allow the ACK time to get going, otherwise the sender sees a failed packet <<<<<<<<<<<<<< ************** !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
198 Radio1.read(&DataPacket, sizeof(DataPacket));
199 if (!IamMaster) { // pupil here
200 MasterDetected(true);
201 if (DataPacket[0] == Pupil_in_Control[0]) {
202 StopBuddyListen(0); // PUPIL -> CONTROL <<<<<< *******
203 DelayWithDog(LONGER_DELAY);
204 return;
205 }
206 }
207 DelayWithDog(LONGER_DELAY);
208 if (TakeControlBackNow) {
209 StopBuddyListen(1); // MASTER RECLAIMS CONTROL <<<<<< *******
210 }
211 LastPassivePacketTime = millis();
212 FlushFifos();
213 }
214 else { // no packet arrived so maybe Pupil's dead, better grab control back
215 if (IamMaster) {
216 if (millis() - LastPassivePacketTime > 700) {
217 StopBuddyListen(1); // MASTER RECLAIMS CONTROL <<<<<< *******
218 }
219 }
220 if (!IamMaster) {
221 if (millis() - LastPassivePacketTime > 700) {
222 MasterDetected(false);
223 LastPassivePacketTime = millis();
224 }
225 }
226 }
