38 if (PupilIsAlive == 2) BuddyON = false; // If pupil is dead, then Buddy is off

277uint8_t EncodeTheChangedChannels1(){
278 #define MIN_CHANGE 4 // Very tiny changes in channel values are ignored. That's most likely only noise... // ... This reduces the average packet size // ... Values <= 20 are imperceptible. So 4 is just fine here.
279 uint8_t NumberOfChangedChannels = 0; // Number of channels that have changed since last packet
280 if ((SpecialPacketData.ResendAllChannels) || (ShorterSpecialPacketData.ResendAllChannels)) // If no ack, then please resend all channels
281 {
282 for (int i = 0; i < CHANNELSUSED; ++i) PreviousBuffer[i] = 0; // Clear the PreviousBuffer when all channels are to be resent
283 SpecialPacketData.ResendAllChannels = false;
284 ShorterSpecialPacketData.ResendAllChannels = false;
285 }
286 DataTosend.ChannelBitMask = 0; // Clear the ChannelBitMask 16 BIT WORD (1 bit per channel)
287 for (int i = 0; i < CHANNELSUSED; ++i){ // Check for changed channels and load them into the rawdatabuffer
288 if ((abs(SendBuffer[i] - PreviousBuffer[i]) >= MIN_CHANGE) && (NumberOfChangedChannels < 4)) // 4 is the maximum number of channel changes that will be sent in one packet ...
289 { // ... any other changes will be sent in the next packet, only 5ms later.
290 RawDataBuffer[NumberOfChangedChannels] = SendBuffer[i]; // Load a changed channel into the rawdatabuffer.
291 PreviousBuffer[i] = SendBuffer[i]; // Save it for next time in case it succeeds this time.
292 DataTosend.ChannelBitMask |= (1 << i); // Set the current bit in the ChannelBitMask word.
293 ++NumberOfChangedChannels; // Increment the number of channel changes (rawdatabuffer index pointer).
294 }
295 }
296 return NumberOfChangedChannels;
297}

299void SendTheSpecialAckPayload(){
300
301 uint8_t ByteCountToTransmit;
302 uint8_t NumberOfChangedChannels = EncodeTheChangedChannels1(); // Encode the changed channels
303
304 if (NumberOfChangedChannels){ // Any channels changed? Or parameters to send?
305 ByteCountToTransmit = ((float) NumberOfChangedChannels * 1.5f) + 4; // Calculate the number of bytes to transmit
306 uint8_t SizeOfUnCompressedData = (ByteCountToTransmit / 1.5) ;
307 Compress(DataTosend.CompressedData, RawDataBuffer, SizeOfUnCompressedData);
308 }else{
309 ByteCountToTransmit = 2;
310 NewCompressNeeded = false;
311 }
312 Radio1.writeAckPayload(1, &DataTosend, ByteCountToTransmit); // Acknowledge the packet BY SENDING MY CHANNEL DATA!
313 delayMicroseconds(30);
314}

334 }

339 Radio1.startListening();

361void GetSpecialPacket(){
362
363 static uint32_t LocalTimer = 0;
364 static uint16_t PacketCounter = 0;
365
366 if (Radio1.available()) { // if a packet has arrived
367 SendTheSpecialAckPayload();
368 uint8_t p = Radio1.getDynamicPayloadSize(); // Get the size of the packet
369
370 if (p > 12){ // p is either 24 or 4
371 ParseLongerSpecialPacket();
372 }else{
373 ParseShorterSpecialPacket();
374 }
375
376 MasterDetected(true); // Master is alive
377 ++PacketCounter; // Count the packets
378
379 if (millis() - LocalTimer >= 1000) { // Every second
380 LocalTimer = millis(); // reset the timer
381 PacketsPerSecond = PacketCounter; // save the packets per second for use on the data diplay
382 PacketCounter = 0; // reset the counter
383 }
384 LastPassivePacketTime = millis(); // reset the timer
385
386 }else{ // No packet arrived so maybe master's dead?
387 if (millis() - LastPassivePacketTime > PACEMAKER) { // We expect a packet every PACEMAKER ( = 5 ) milliseconds. If none, a packet was lost.
388 Radio1.stopListening();
389 Radio1.setChannel(QUIETCHANNEL); // Set the recovery channel
390 Radio1.startListening();
391 MasterDetected(false);
392 LastPassivePacketTime = millis(); // Reset the timer
393 ++TotalLostPackets; // Count the lost packets
394 }
395 }
396}
