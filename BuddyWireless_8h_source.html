19void LoadCorrectModel(uint64_t ModelID){ // Gets Buddy onto same model as master, quietly.
20 uint32_t SavedModelNumber = ModelNumber; // Save the current model number
21 ModelMatched = false;
22 ModelNumber = 0;

35 GotoFrontView();
36 }
37}

40void GetSlaveChannelValuesWireless(){ // Very Like the PPM function only a bit simpler
41
42 if (PupilIsAlive == 2) BuddyON = false; // If pupil is dead, then Buddy is off
43 if (BuddyON) {
44 for (int j = 0; j < CHANNELSUSED; ++j) { // While slave has control, his stick data replaces some of ours
45 if (BuddyControlled & (1 << j)) SendBuffer[j] = BuddyBuffer[j]; // Test if this channel is buddy controlled. If not leave it unchanged otherwise replace it with the buddy's value
46 }
47 if (!SlaveHasControl) { // Buddy has turned On
48 PlaySound(BUDDYMSG); // Announce the Buddy is now in control
49 SlaveHasControl = true;
50 }
51 }
52 else {
53 if (SlaveHasControl) { // Buddy has turned Off
54 PlaySound(MASTERMSG); // Announce the Master is now in control
55 SlaveHasControl = false;
56 }
57 }

122//*************************************************************************************************************************
123// Master gets Ack from pupil.
124// Pupil's Ack Payload contains the pupil's control data (in BuddyBuffer[]) which is compressed and must be decompressed before use.
125// This function is called from SendSpecialPacket() which is called from the main loop.
126// GetSlaveChannelValuesWireless() then uses the BuddyBuffer data to replace some or all of the Master's control data.
127void GetPupilAck()
128{
129 uint16_t AckSpecial[COMPRESSEDWORDS]; // Ack payload will contain all the pupil's control data
130 if (Radio1.available()) { // if a packet has arrived
131 Radio1.read(&AckSpecial, sizeof AckSpecial); // read the packet
132 Decompress(BuddyBuffer, AckSpecial, UNCOMPRESSEDWORDS); // decompress the data into buddybuffer array
133 }
134}

137void SendSpecialPacket() // Here the MASTER sends to PUPIL tx. This is called about 100 times a second.
138{ // Master Sends M or B to indicate whether Buddy is on or off, and the ID of the model which should be loaded.
139 struct spd
140 {
141 char Command[2];
142 uint64_t ModelID;
143 };
144 spd SpecialPacketData;
145
146 SpecialPacketData.ModelID = ModelsMacUnionSaved.Val64; // send the model ID so that pupil can check it
147 SpecialPacketData.Command[0] = 'M'; // Send M to indicate Master is ON
148 if (BuddyON) SpecialPacketData.Command[0] = 'B'; // Send B to indicate Buddy is ON
149 Radio1.openWritingPipe(TeensyMACAddPipe ^ ENCRYPT_KEY); // send to encrypted pipe address
150 Radio1.setDataRate(FASTDATARATE); // 2MBPS
151 Radio1.setChannel(SPECIAL_PACKET_CHANNEL);
152 if (Radio1.write(&SpecialPacketData, sizeof SpecialPacketData)) {
153 GetPupilAck(); // get ack from pupil WITH HIS CONTROL DATA!!
154 PupilDetected(true); // pupil is alive
155 } else {
156 PupilDetected(false); // pupil is dead
157 }
158 Radio1.setDataRate(DATARATE); // restore the proper data rate
159 Radio1.openWritingPipe(TeensyMACAddPipe); // restore the proper pipe address
160 Radio1.setChannel(CurrentChannel); // restore the proper frequency channel
161}

173void GetSpecialPacket() // here the PUPIL tx gets from MASTER tx. This function is called from main loop about 100 times a second.
174{ // Master tells Pupil whether buddy is on or off, and the ID of the model which should be loaded.
175 static bool MasterIsInControl = true;
176
177 struct spd {
178 char Command[2];
179 uint64_t ModelID = 0;
180 };
181 spd SpecialPacketData;
182
183 if (Radio1.available()) { // if a packet has arrived
184 Radio1.writeAckPayload(1, &DataTosend.CompressedData, SizeOfCompressedData); // Acknowledge the packet BY SENDING MY CHANNEL DATA!
185 DelayWithDog(DELAYAFTERACK); // <- ** MUST ** allow the ACK time to get going, otherwise the sender sees a failed packet
186 if (Radio1.available()){
187 Radio1.read(&SpecialPacketData, sizeof SpecialPacketData); // read the packet if its still there
188 if ((SpecialPacketData.Command[0] == 'B') && (MasterIsInControl)) { // Buddy is now in control
189 MasterIsInControl = false; // Buddy is now in control
190 PlaySound(BUDDYMSG); // Announce the Buddy is now in control
191 }
192 if ((SpecialPacketData.Command[0] == 'M') && (!MasterIsInControl)) { // Master is now in control
193 MasterIsInControl = true; // Master is now in control
194 PlaySound(MASTERMSG); // Announce the Master is now in control
195 }
196 CheckModelMatchesMaster(SpecialPacketData.ModelID); // check the model ID (once every second) and load the correct model if it !matches
197 MasterDetected(true); // Master is alive
198 LastPassivePacketTime = millis(); // reset the timer
199 FlushFifos(); // flush the fifos
200 }
201 }else{ // no packet arrived so maybe master's dead
202
203 if (millis() - LastPassivePacketTime > 700) {
204 MasterDetected(false);
205 LastPassivePacketTime = millis();
206 }
207 }
208}

211void StartBuddyListen()
212{
213 Radio1.setDataRate(FASTDATARATE); // 2 MBPS
214 Radio1.enableAckPayload(); // needed
215 Radio1.enableDynamicPayloads(); // needed
216 Radio1.setAutoAck(true); // we want acks
217 Radio1.maskIRQ(1, 1, 1); // no interrupts - seems NEEDED at the moment
218 Radio1.openReadingPipe(1, BuddyMACAddPipe ^ ENCRYPT_KEY);
219 delayMicroseconds(SHORT_DELAY); // to allow the pipe to open
220 Radio1.setChannel(SPECIAL_PACKET_CHANNEL); // set the channel to the special packet channel
221 Radio1.startListening(); // start listening
222 FlushFifos(); // flush the fifos
223 BlueLedOn(); // turn on the blue led
224 CurrentMode = LISTENMODE; // set the mode to listen
225 //LostContactFlag = false; // reset the lost contact flag
226 RestoreBrightness(); // restore the brightness
227}
