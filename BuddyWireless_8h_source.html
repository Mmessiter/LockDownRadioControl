38 if (PupilIsAlive == 2) BuddyON = false; // If pupil is dead, then Buddy is off
39 if (BuddyON) {

122// Master gets Ack from pupil.
123// Pupil's Ack Payload contains the pupil's control data (in BuddyBuffer[]) which is compressed and must be decompressed before use.
124// This function is called from SendSpecialPacket() which is called from the main loop.
125// GetSlaveChannelValuesWireless() then uses the BuddyBuffer data to replace some or all of the Master's control data.
126void GetPupilAck()
127{
128 uint16_t AckSpecial[COMPRESSEDWORDS]; // Ack payload will contain all the pupil's control data
129 if (Radio1.available()) { // if a packet has arrived
130 Radio1.read(&AckSpecial, sizeof AckSpecial); // read the packet
131 Decompress(BuddyBuffer, AckSpecial, UNCOMPRESSEDWORDS); // decompress the data into buddybuffer array
132 }

147void SendSpecialPacket() // Here the MASTER sends to PUPIL tx. This is called about 200 times a second.
148{ // Master Sends M or B to indicate whether Buddy is on or off, and the ID of the model which should be loaded.
149 // The Pupil's Ack Payload contains the pupil's control data (in BuddyBuffer[]) which is compressed and must be decompressed before use.
150 // This function is called from SendData() function which is called from the main loop.
151 // GetSlaveChannelValuesWireless() then uses the BuddyBuffer data to replace some or all of the Master's control data.
152 // Because the datarate is 2 meg the exchange is very fast.
153 static uint32_t LocalTimer = 0;
154 static bool NeedToRecover = false;
155 static uint8_t ChannelSentLastTime = 0; // The old channel number
156 static uint8_t Index = 82; // The current channel number
157 struct spd
158 {
159 char Command[2];
160 uint64_t ModelID;
161 uint8_t Channel = QUIETCHANNEL;
162 };
163 static spd SpecialPacketData;
164
165 rf24_datarate_e faster = FASTDATARATE, slower = DATARATE; // Save the data rates
166 if ((!BoundFlag || !ModelMatched || (PupilIsAlive != 1))){ // Don't send too often if not connected, nor if buddy was not found
167 if (((millis() - LocalTimer) < 20)) return;
168 LocalTimer = millis();
169 }
170 SpecialPacketData.ModelID = ModelsMacUnionSaved.Val64; // Send the model ID so that pupil can check it
171 SpecialPacketData.Command[0] = 'M'; // Send M to indicate Master is ON
172 if (BuddyON) SpecialPacketData.Command[0] = 'B'; // Send B to indicate Buddy is ON
173 ChannelSentLastTime = SpecialPacketData.Channel; // Use the old channel number because Buddy hasn't yet hopped
174 --Index; if (Index < 1) Index = 82; // use the same array but in reverse order
175 SpecialPacketData.Channel = FHSS_data::FHSS_Channels[Index]; // Set the new channel number for next time
176 if (NeedToRecover) SpecialPacketData.Channel = QUIETCHANNEL; // If contact lost, then use the recovery channel to recover
177 ChangeTXTarget(ChannelSentLastTime,TeensyMACAddPipe ^ ENCRYPT_KEY,faster); // Set the TX target to the Buddy
178 if (Radio1.write(&SpecialPacketData, sizeof SpecialPacketData)) { // Send the packet
179 GetPupilAck(); // Get ack from pupil WITH HIS CONTROL DATA!!
180 PupilDetected(true); // Pupil is alive
181 NeedToRecover = false; // No need to recover
182 } else {
183 PupilDetected(false); // Pupil is dead
184 NeedToRecover = true; // Need to recover
185 }
186 ChangeTXTarget(CurrentChannel,TeensyMACAddPipe,slower); // Set the TX target back to the receiver in model.
187}

246void GetSpecialPacket() // Here the PUPIL tx gets a packet from MASTER tx. This function is called from main loop very fast.
247{ // Master tells Pupil whether buddy is on or off, and the ID of the model which should be loaded.
248 // The Pupil's Ack Payload contains the pupil's control data (in BuddyBuffer[]) which is compressed and must be decompressed before use.
249 // GetSlaveChannelValuesWireless() in Master then uses the BuddyBuffer data to replace some or all of the Master's control data.
250 // Because the datarate is 2 meg the exchange is very fast.
251 static bool MasterIsInControl = true;
252 static uint16_t PacketCounter = 0;
253 static uint32_t LocalTimer = 0;
254 static uint32_t LastModelID = 0;
255 struct spd {
256 char Command[2];
257 uint64_t ModelID = 0;
258 uint8_t Channel = QUIETCHANNEL;
259
260 };
261 spd SpecialPacketData;
262 Compress(DataTosend.CompressedData, SendBuffer, UNCOMPRESSEDWORDS); // Compress
263 if (Radio1.available()) { // if a packet has arrived
264 Radio1.writeAckPayload(1, &DataTosend.CompressedData, SizeOfCompressedData); // Acknowledge the packet BY SENDING MY CHANNEL DATA!
265 delay(1); // <- *MUST* allow the ACK time to get going, otherwise the sender sees a failed packet
266 Radio1.read(&SpecialPacketData, sizeof SpecialPacketData); // read the packet if its still there
267 if ((SpecialPacketData.Command[0] == 'B') && (MasterIsInControl)) { // Buddy is now in control
268 MasterIsInControl = false; // Buddy is now in control
269 PlaySound(BUDDYMSG); // Announce the Buddy is now in control
270 }
271 if ((SpecialPacketData.Command[0] == 'M') && (!MasterIsInControl)) { // Master is now in control
272 MasterIsInControl = true; // Master is now in control
273 PlaySound(MASTERMSG); // Announce the Master is now in control
274 }
275 if (SpecialPacketData.ModelID != LastModelID) { // if the model ID has changed, match it if we can
276 LastModelID = SpecialPacketData.ModelID; // Save the model ID so we can know if it changed
277 if (SpecialPacketData.ModelID != ModelsMacUnionSaved.Val64){ // is it the currently loaded model?
278 LoadCorrectModel(SpecialPacketData.ModelID); // if not then try only once to load the correct model
279 }
280 }
281 MasterDetected(true); // Master is alive
282 ++PacketCounter; // Count the packets
283 if (millis() - LocalTimer > 1000) { // Every second
284 LocalTimer = millis(); // reset the timer
285 PacketsPerSecond = PacketCounter; // save the packets per second for use on the data diplay
286 PacketCounter = 0; // reset the counter
287 }
288 LastPassivePacketTime = millis(); // reset the timer
289 Radio1.stopListening();
290 Radio1.setChannel(SpecialPacketData.Channel); // Set the frequency channel
291 Radio1.startListening(); // Start listening
292 }else{ // No packet arrived so maybe master's dead?
293 if (millis() - LastPassivePacketTime > PACEMAKER) { // We expect a packet every PACEMAKER ( = 5 ) milliseconds. If none, a packet was lost.
294 Radio1.stopListening();
295 Radio1.setChannel(QUIETCHANNEL); // Set the recovery channel
296 Radio1.startListening();
297 MasterDetected(false);
298 LastPassivePacketTime = millis(); // Reset the timer
299 ++TotalLostPackets; // Count the lost packets
300 }
301 }
302}

305void StartBuddyListen()
306{
307 Radio1.setDataRate(FASTDATARATE); // 2 MBPS
308 Radio1.enableAckPayload(); // needed
309 Radio1.enableDynamicPayloads(); // needed
310 Radio1.setAutoAck(true); // we want acks

313 delayMicroseconds(STOPLISTENINGDELAY); // to allow the pipe to open
314 Radio1.setChannel(QUIETCHANNEL); // set the channel to the recovery channel
315 Radio1.startListening(); // start listening
316 FlushFifos(); // flush the fifos
317 BlueLedOn(); // turn on the blue led
318 CurrentMode = LISTENMODE; // set the mode to listen
319 WasBuddyPupilOnWireless = true; // flag to indicate that the buddy was on wireless
