11void StartMacro(uint8_t m)
12{ // Start a macro
13 MacrosBuffer[m][MACRORUNNINGNOW] |= 1; // LOW BIT = "running now" flag

18void RunMacro(uint8_t m)
19{ // Move a servo to a place
20 uint32_t RightNow = millis();
21 if (RightNow >= MacroStartTime[m]) MacrosBuffer[m][MACRORUNNINGNOW] |= 2; // Set the ACTIVE Bit if started (BIT 1)
22 if (RightNow >= MacroStopTime[m]) MacrosBuffer[m][MACRORUNNINGNOW] &= 1; // Clear the ACTIVE Bit if expired (BIT 1)
23 if (MacrosBuffer[m][MACRORUNNINGNOW] & 2) {
24 SendBuffer[(MacrosBuffer[m][MACROMOVECHANNEL]) - 1] = map(MacrosBuffer[m][MACROMOVETOPOSITION], 0, 180, MINMICROS, MAXMICROS); // Do it if currently active!
25 }
26}

34void ExecuteMacro()
35{ // Main entry point from SendData() ... START/STOP/RUN
36 uint8_t TriggerChannel = 0;
37 for (u_int8_t i = 0; i < MAXMACROS; ++i) {
38 // ***************************** START OR STOP ******************************
39 TriggerChannel = (MacrosBuffer[i][MACROTRIGGERCHANNEL]) - 1; // Down by one as channels are really 0 - 15
40 if (TriggerChannel) { // Is trigger channel non-zero?
41 if (SendBuffer[TriggerChannel] >= MAXMICROS - 1) { // Is the trigger point is close to its highest value?
42 if (!MacrosBuffer[i][MACRORUNNINGNOW]) StartMacro(i); // Yes. Start if not already started
43 }
44 else {
45 if (MacrosBuffer[i][MACRORUNNINGNOW]) StopMacro(i); // No. Stop it if it was running
46 }
47 // **************************** RUN ****************************************
48 if (MacrosBuffer[i][MACRORUNNINGNOW]) { // If running, move the servo ... if timer agrees.
49 RunMacro(i);

51 }
