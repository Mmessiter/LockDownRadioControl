277float CalculateRate(short int Curve, short int OutputChannel, float rate)
278{
279
280 switch (Curve) { // Curve is 1 - 5, low to hi.
281 case 0:
282 return (((MinDegrees[Bank][OutputChannel]) - 90) * (rate / 100)) + 90;
283 case 1:
284 return (((MidLowDegrees[Bank][OutputChannel]) - 90) * (rate / 100)) + 90;
285 case 2:
286 return (((CentreDegrees[Bank][OutputChannel]) - 90) * (rate / 100)) + 90;
287 case 3:
288 return (((MidHiDegrees[Bank][OutputChannel]) - 90) * (rate / 100)) + 90;
289 case 4:
290 return (((MaxDegrees[Bank][OutputChannel]) - 90) * (rate / 100)) + 90;
291 default:
292 return 0;
293 }
294}
