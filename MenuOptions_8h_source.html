377// This function receives upto 50 data elements from the Nextion display and loads it into ScreenData array of uint16_t
378
379void ReceiveLotsofData(){
380 int i = 0;
381 union{
382 uint8_t First4Bytes[4];
383 uint32_t FirstDWord;
384 }
385 NextionData;
386
387 for (int field = 1; field < 49; ++field){
388 int offset = field * 4;
389 for (int p = 0; p < 4; ++p) NextionData.First4Bytes[p] = TextIn[offset+p];
390 if (NextionData.FirstDWord < 0xFFFF){
391 ScreenData[i] = NextionData.FirstDWord;
392 ++i;
393 }else{
394 break;
395 }
396 }
397 switch (CurrentView) {
398 case MIXESVIEW:
399 ReadMixValues();
400 break;
401 case DUALRATESVIEW:
402 DualRatesRefresh();
403 break;
404 default:
405 break;
406 }
407
408}
