443void getIMUdata() {
444
445 int16_t AcX,AcY,AcZ,GyX,GyY,GyZ;
446 mpu6050.getMotion6(&AcX, &AcY, &AcZ, &GyX, &GyY, &GyZ);
447
448 //Accelerometer
449 AccX = AcX / ACCEL_SCALE_FACTOR; //G's
450 AccY = AcY / ACCEL_SCALE_FACTOR;
451 AccZ = AcZ / ACCEL_SCALE_FACTOR;
452 //Correct the outputs with the calculated error values
453 AccX = AccX - AccErrorX;
454 AccY = AccY - AccErrorY;
455 AccZ = AccZ - AccErrorZ;
456 //LP filter accelerometer data
457 AccX = (1.0 - B_accel)*AccX_prev + B_accel*AccX;
458 AccY = (1.0 - B_accel)*AccY_prev + B_accel*AccY;
459 AccZ = (1.0 - B_accel)*AccZ_prev + B_accel*AccZ;
460 AccX_prev = AccX;
461 AccY_prev = AccY;
462 AccZ_prev = AccZ;
463
464 //Gyro
465 GyroX = GyX / GYRO_SCALE_FACTOR; //deg/sec
466 GyroY = GyY / GYRO_SCALE_FACTOR;
467 GyroZ = GyZ / GYRO_SCALE_FACTOR;
468 //Correct the outputs with the calculated error values
469 GyroX = GyroX - GyroErrorX;
470 GyroY = GyroY - GyroErrorY;
471 GyroZ = GyroZ - GyroErrorZ;
472 //LP filter gyro data
473 GyroX = (1.0 - B_gyro)*GyroX_prev + B_gyro*GyroX;
474 GyroY = (1.0 - B_gyro)*GyroY_prev + B_gyro*GyroY;
475 GyroZ = (1.0 - B_gyro)*GyroZ_prev + B_gyro*GyroZ;
476 GyroX_prev = GyroX;
477 GyroY_prev = GyroY;
478 GyroZ_prev = GyroZ;
479
480}

484void calculate_IMU_error() {
485
486 int16_t AcX,AcY,AcZ,GyX,GyY,GyZ;
487
488 //Read IMU values 12000 times
489 int c = 0;
490 while (c < 12000) {
491 mpu6050.getMotion6(&AcX, &AcY, &AcZ, &GyX, &GyY, &GyZ);
492 AccX = AcX / ACCEL_SCALE_FACTOR;
493 AccY = AcY / ACCEL_SCALE_FACTOR;
494 AccZ = AcZ / ACCEL_SCALE_FACTOR;
495 GyroX = GyX / GYRO_SCALE_FACTOR;
496 GyroY = GyY / GYRO_SCALE_FACTOR;
497 GyroZ = GyZ / GYRO_SCALE_FACTOR;
498
499 //Sum all readings
500 AccErrorX = AccErrorX + AccX;
501 AccErrorY = AccErrorY + AccY;
502 AccErrorZ = AccErrorZ + AccZ;
503 GyroErrorX = GyroErrorX + GyroX;
504 GyroErrorY = GyroErrorY + GyroY;
505 GyroErrorZ = GyroErrorZ + GyroZ;
506 c++;
507 }
508 //Divide the sum by 12000 to get the error value
509 AccErrorX = AccErrorX / c;
510 AccErrorY = AccErrorY / c;
511 AccErrorZ = AccErrorZ / c - 1.0;
512 GyroErrorX = GyroErrorX / c;
513 GyroErrorY = GyroErrorY / c;
514 GyroErrorZ = GyroErrorZ / c;
515}

535 float recipNorm;
536 float s0, s1, s2, s3;
537 float qDot1, qDot2, qDot3, qDot4;
538 float hx, hy;
539 float _2q0mx, _2q0my, _2q0mz, _2q1mx, _2bx, _2bz, _4bx, _4bz, _2q0, _2q1, _2q2, _2q3, _2q0q2, _2q2q3, q0q0, q0q1, q0q2, q0q3, q1q1, q1q2, q1q3, q2q2, q2q3, q3q3;
540
541 Madgwick6DOF(gx, gy, gz, ax, ay, az, invSampleFreq);
542 return;
543
544 if((mx == 0.0f) && (my == 0.0f) && (mz == 0.0f)) {
545 Madgwick6DOF(gx, gy, gz, ax, ay, az, invSampleFreq);
546 return;
547 }
548
549 gx *= 0.0174533f;
550 gy *= 0.0174533f;
551 gz *= 0.0174533f;
552
553 qDot1 = 0.5f * (-q1 * gx - q2 * gy - q3 * gz);
554 qDot2 = 0.5f * (q0 * gx + q2 * gz - q3 * gy);
555 qDot3 = 0.5f * (q0 * gy - q1 * gz + q3 * gx);
556 qDot4 = 0.5f * (q0 * gz + q1 * gy - q2 * gx);
557
558
559 if(!((ax == 0.0f) && (ay == 0.0f) && (az == 0.0f))) {
560
561
562 recipNorm = invSqrt(ax * ax + ay * ay + az * az);
563 ax *= recipNorm;
564 ay *= recipNorm;
565 az *= recipNorm;
566
567
568 recipNorm = invSqrt(mx * mx + my * my + mz * mz);
569 mx *= recipNorm;
570 my *= recipNorm;
571 mz *= recipNorm;
572
573
574 _2q0mx = 2.0f * q0 * mx;
575 _2q0my = 2.0f * q0 * my;
576 _2q0mz = 2.0f * q0 * mz;
577 _2q1mx = 2.0f * q1 * mx;
578 _2q0 = 2.0f * q0;
579 _2q1 = 2.0f * q1;
580 _2q2 = 2.0f * q2;
581 _2q3 = 2.0f * q3;
582 _2q0q2 = 2.0f * q0 * q2;
583 _2q2q3 = 2.0f * q2 * q3;
584 q0q0 = q0 * q0;
585 q0q1 = q0 * q1;
586 q0q2 = q0 * q2;
587 q0q3 = q0 * q3;
588 q1q1 = q1 * q1;
589 q1q2 = q1 * q2;
590 q1q3 = q1 * q3;
591 q2q2 = q2 * q2;
592 q2q3 = q2 * q3;
593 q3q3 = q3 * q3;
594
595 //Reference direction of Earth's magnetic field
596 hx = mx * q0q0 - _2q0my * q3 + _2q0mz * q2 + mx * q1q1 + _2q1 * my * q2 + _2q1 * mz * q3 - mx * q2q2 - mx * q3q3;
597 hy = _2q0mx * q3 + my * q0q0 - _2q0mz * q1 + _2q1mx * q2 - my * q1q1 + my * q2q2 + _2q2 * mz * q3 - my * q3q3;
598 _2bx = sqrtf(hx * hx + hy * hy);
599 _2bz = -_2q0mx * q2 + _2q0my * q1 + mz * q0q0 + _2q1mx * q3 - mz * q1q1 + _2q2 * my * q3 - mz * q2q2 + mz * q3q3;
600 _4bx = 2.0f * _2bx;
601 _4bz = 2.0f * _2bz;
602
603 //Gradient decent algorithm corrective step
604 s0 = -_2q2 * (2.0f * q1q3 - _2q0q2 - ax) + _2q1 * (2.0f * q0q1 + _2q2q3 - ay) - _2bz * q2 * (_2bx * (0.5f - q2q2 - q3q3) + _2bz * (q1q3 - q0q2) - mx) + (-_2bx * q3 + _2bz * q1) * (_2bx * (q1q2 - q0q3) + _2bz * (q0q1 + q2q3) - my) + _2bx * q2 * (_2bx * (q0q2 + q1q3) + _2bz * (0.5f - q1q1 - q2q2) - mz);
605 s1 = _2q3 * (2.0f * q1q3 - _2q0q2 - ax) + _2q0 * (2.0f * q0q1 + _2q2q3 - ay) - 4.0f * q1 * (1 - 2.0f * q1q1 - 2.0f * q2q2 - az) + _2bz * q3 * (_2bx * (0.5f - q2q2 - q3q3) + _2bz * (q1q3 - q0q2) - mx) + (_2bx * q2 + _2bz * q0) * (_2bx * (q1q2 - q0q3) + _2bz * (q0q1 + q2q3) - my) + (_2bx * q3 - _4bz * q1) * (_2bx * (q0q2 + q1q3) + _2bz * (0.5f - q1q1 - q2q2) - mz);
606 s2 = -_2q0 * (2.0f * q1q3 - _2q0q2 - ax) + _2q3 * (2.0f * q0q1 + _2q2q3 - ay) - 4.0f * q2 * (1 - 2.0f * q1q1 - 2.0f * q2q2 - az) + (-_4bx * q2 - _2bz * q0) * (_2bx * (0.5f - q2q2 - q3q3) + _2bz * (q1q3 - q0q2) - mx) + (_2bx * q1 + _2bz * q3) * (_2bx * (q1q2 - q0q3) + _2bz * (q0q1 + q2q3) - my) + (_2bx * q0 - _4bz * q2) * (_2bx * (q0q2 + q1q3) + _2bz * (0.5f - q1q1 - q2q2) - mz);
607 s3 = _2q1 * (2.0f * q1q3 - _2q0q2 - ax) + _2q2 * (2.0f * q0q1 + _2q2q3 - ay) + (-_4bx * q3 + _2bz * q1) * (_2bx * (0.5f - q2q2 - q3q3) + _2bz * (q1q3 - q0q2) - mx) + (-_2bx * q0 + _2bz * q2) * (_2bx * (q1q2 - q0q3) + _2bz * (q0q1 + q2q3) - my) + _2bx * q1 * (_2bx * (q0q2 + q1q3) + _2bz * (0.5f - q1q1 - q2q2) - mz);
608 recipNorm = invSqrt(s0 * s0 + s1 * s1 + s2 * s2 + s3 * s3); // normalise step magnitude
609 s0 *= recipNorm;
610 s1 *= recipNorm;
611 s2 *= recipNorm;
612 s3 *= recipNorm;
613
614 //Apply feedback step
615 qDot1 -= B_madgwick * s0;
616 qDot2 -= B_madgwick * s1;
617 qDot3 -= B_madgwick * s2;
618 qDot4 -= B_madgwick * s3;
619 }
620
621 //Integrate rate of change of quaternion to yield quaternion
622 q0 += qDot1 * invSampleFreq;
623 q1 += qDot2 * invSampleFreq;
624 q2 += qDot3 * invSampleFreq;
625 q3 += qDot4 * invSampleFreq;
626
627 //Normalise quaternion
628 recipNorm = invSqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
629 q0 *= recipNorm;
630 q1 *= recipNorm;
631 q2 *= recipNorm;
632 q3 *= recipNorm;
633
634 //compute angles - NWU
635 roll_IMU = atan2(q0*q1 + q2*q3, 0.5f - q1*q1 - q2*q2)*57.29577951; //degrees
636 pitch_IMU = -asin(-2.0f * (q1*q3 - q0*q2))*57.29577951; //degrees
637 yaw_IMU = -atan2(q1*q2 + q0*q3, 0.5f - q2*q2 - q3*q3)*57.29577951; //degrees
638}

644 float recipNorm;
645 float s0, s1, s2, s3;
646 float qDot1, qDot2, qDot3, qDot4;
647 float _2q0, _2q1, _2q2, _2q3, _4q0, _4q1, _4q2 ,_8q1, _8q2, q0q0, q1q1, q2q2, q3q3;
648
649
650 gx *= 0.0174533f;
651 gy *= 0.0174533f;
652 gz *= 0.0174533f;
653
654
655 qDot1 = 0.5f * (-q1 * gx - q2 * gy - q3 * gz);
656 qDot2 = 0.5f * (q0 * gx + q2 * gz - q3 * gy);
657 qDot3 = 0.5f * (q0 * gy - q1 * gz + q3 * gx);
658 qDot4 = 0.5f * (q0 * gz + q1 * gy - q2 * gx);
659
660
661 if(!((ax == 0.0f) && (ay == 0.0f) && (az == 0.0f))) {
662
663 recipNorm = invSqrt(ax * ax + ay * ay + az * az);
664 ax *= recipNorm;
665 ay *= recipNorm;
666 az *= recipNorm;
667
668
669 _2q0 = 2.0f * q0;
670 _2q1 = 2.0f * q1;
671 _2q2 = 2.0f * q2;
672 _2q3 = 2.0f * q3;
673 _4q0 = 4.0f * q0;
674 _4q1 = 4.0f * q1;
675 _4q2 = 4.0f * q2;
676 _8q1 = 8.0f * q1;
677 _8q2 = 8.0f * q2;
678 q0q0 = q0 * q0;
679 q1q1 = q1 * q1;
680 q2q2 = q2 * q2;
681 q3q3 = q3 * q3;
682

686 s2 = 4.0f * q0q0 * q2 + _2q0 * ax + _4q2 * q3q3 - _2q3 * ay - _4q2 + _8q2 * q1q1 + _8q2 * q2q2 + _4q2 * az;
687 s3 = 4.0f * q1q1 * q3 - _2q1 * ax + 4.0f * q2q2 * q3 - _2q2 * ay;
688 recipNorm = invSqrt(s0 * s0 + s1 * s1 + s2 * s2 + s3 * s3);
689 s0 *= recipNorm;
690 s1 *= recipNorm;
691 s2 *= recipNorm;
692 s3 *= recipNorm;
693
694
695 qDot1 -= B_madgwick * s0;
696 qDot2 -= B_madgwick * s1;
697 qDot3 -= B_madgwick * s2;
698 qDot4 -= B_madgwick * s3;
699 }
700
701
702 q0 += qDot1 * invSampleFreq;
703 q1 += qDot2 * invSampleFreq;
704 q2 += qDot3 * invSampleFreq;
705 q3 += qDot4 * invSampleFreq;
706
707
708 recipNorm = invSqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
709 q0 *= recipNorm;
710 q1 *= recipNorm;
711 q2 *= recipNorm;
712 q3 *= recipNorm;
713
714
715 roll_IMU = atan2(q0*q1 + q2*q3, 0.5f - q1*q1 - q2*q2)*57.29577951; //degrees
716 pitch_IMU = -asin(-2.0f * (q1*q3 - q0*q2))*57.29577951; //degrees
717 yaw_IMU = -atan2(q1*q2 + q0*q3, 0.5f - q2*q2 - q3*q3)*57.29577951; //degrees
718}

1147float floatFaderLinear(float param, float param_min, float param_max, float fadeTime, int state, int loopFreq){
1148 //DESCRIPTION: Linearly fades a float type variable between min and max bounds based on desired high or low state and time
1149 /*
1150 * Takes in a float variable, desired minimum and maximum bounds, fade time, high or low desired state, and the loop frequency
1151 * and linearly interpolates that param variable between the maximum and minimum bounds. This function can be called in controlMixer()
1152 * and high/low states can be determined by monitoring the state of an auxillarly radio channel. For example, if channel_6_pwm is being
1153 * monitored to switch between two dynamic configurations (hover and forward flight), this function can be called within the logical
1154 * statements in order to fade controller gains, for example between the two dynamic configurations. The 'state' (1 or 0) can be used
1155 * to designate the two final options for that control gain based on the dynamic configuration assignment to the auxillary radio channel.
1156 *
1157 */
1158 float diffParam = (param_max - param_min)/(fadeTime*loopFreq); //difference to add or subtract from param for each loop iteration for desired fadeTime
1159
1160 if (state == 1) { //maximum param bound desired, increase param by diffParam for each loop iteration
1161 param = param + diffParam;
1162 }
1163 else if (state == 0) { //minimum param bound desired, decrease param by diffParam for each loop iteration
1164 param = param - diffParam;
1165 }
1166 param = constrain(param, param_min, param_max); //constrain param within max bounds
1167 return param;
1168}
