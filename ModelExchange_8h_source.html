149 Fbuffer[BUFFERSIZE] = Fsize;
150 Fbuffer[BUFFERSIZE + 1] = Fsize >> 8;
151 Fbuffer[BUFFERSIZE + 2] = Fsize >> 16;
152 Fbuffer[BUFFERSIZE + 3] = Fsize >> 24; // SEND FILE SIZE (four bytes)
153 for (int q = 0; q < 5; ++q) {
154 Fbuffer[q + 16] = MacAddress[q + 1]; // Add only 5 bytes of the macaddress to buffer at offset 16 and sent it too! heer
155 }
156 }
157 else {
158 ModelsFileNumber.seek(Fposition); // Move filepointer
159 ShortDelay();
160 ModelsFileNumber.read(Fbuffer, BUFFERSIZE); // Read part of file
161 Fposition += BUFFERSIZE;
162 if (Fposition > Fsize) Fposition = Fsize;
163 }
164 Radio1.write(&Fbuffer, BUFFERSIZE + 4); // we need error checking!!!
165 Radio1.read(&Fack, sizeof(Fack));
166
167 #ifdef DB_MODEL_EXCHANGE
168 Serial.println(PacketNumber);
169 #endif
170 }
171 ModelsFileNumber.close();
172 #ifdef DB_MODEL_EXCHANGE
173 Serial.println("ALL SENT.");
174 #endif
175 SendValue(Progress, 100);
176 if (ReceiverConnected) DelayWithDog(750);
177 NormaliseTheRadio();
178 SendCommand(ProgressEnd);
179 RedLedOn();
180 if (ReceiverConnected) {
181 strcpy(msg, Sent);
182 strcat(msg, Str(nb1, Fsize, 0));
183 strcat(msg, bytes);
184 PlaySound(BEEPCOMPLETE);
185 ShowFileProgress(msg);
186 DelayWithDog(2000);
187 }
188 else {
189 strcpy(msg, "Receiving TX not ready!");
190 ShowFileProgress(msg);
191 for (int i = 0; i < 3; ++i) {
192 PlaySound(BEEPMIDDLE);
193 DelayWithDog(130);
194 }
195 DelayWithDog(610);
196 }
197 SendCommand(GoModelsView);
198 CurrentView = MODELSVIEW;
199 CloseModelsFile();
200 ConfigureRadio();
201}
