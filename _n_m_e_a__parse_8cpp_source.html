551 /**************************************************************************/
559 /**************************************************************************/
560 bool Adafruit_GPS::check(char *nmea) {
561  thisCheck = 0; // new check
562  *thisSentence = *thisSource = 0;
563  if (*nmea != '$' && *nmea != '!')
564  return false; // doesn't start with $ or !
565  else
566  thisCheck += NMEA_HAS_DOLLAR;
567  // do checksum check -- first look if we even have one -- ignore all but last
568  // *
569  char *ast = nmea; // not strchr(nmea,'*'); for first *
570  while (*ast)
571  ast++; // go to the end
572  while (*ast != '*' && ast > nmea)
573  ast--; // then back to * if it's there
574  if (*ast != '*')
575  return false; // there is no asterisk
576  else {
577  uint16_t sum = parseHex(*(ast + 1)) * 16; // extract checksum
578  sum += parseHex(*(ast + 2));
579  char *p = nmea; // check checksum
580  for (char *p1 = p + 1; p1 < ast; p1++)
581  sum ^= *p1;
582  if (sum != 0)
583  return false; // bad checksum :(
584  else
585  thisCheck += NMEA_HAS_CHECKSUM;
586  }
587  // extract source of variable length
588  char *p = nmea + 1;
589  const char *src = tokenOnList(p, sources);
590  if (src) {
591  strcpy(thisSource, src);
592  thisCheck += NMEA_HAS_SOURCE;
593  } else
594  return false;
595  p += strlen(src);
596  // extract sentence id and check if parsed
597  const char *snc = tokenOnList(p, sentences_parsed);
598  if (snc) {
599  strcpy(thisSentence, snc);
600  thisCheck += NMEA_HAS_SENTENCE_P + NMEA_HAS_SENTENCE;
601  } else { // check if known
602  snc = tokenOnList(p, sentences_known);
603  if (snc) {
604  strcpy(thisSentence, snc);
605  thisCheck += NMEA_HAS_SENTENCE;
606  return false; // known but not parsed
607  } else {
608  parseStr(thisSentence, p, NMEA_MAX_SENTENCE_ID);
609  return false; // unknown
610  }
611  }
612  return true; // passed all the tests
613 }

624  int i = 0; // index in the list
625  while (strncmp(list[i], "ZZ", 2) &&
626  i < 1000) { // stop at terminator and don't crash without it
627  // test for a match on the sentence name
628  if (!strncmp((const char *)list[i], (const char *)token, strlen(list[i])))

630  i++;
631  }

656 /**************************************************************************/
678 /**************************************************************************/
679 bool Adafruit_GPS::parseCoord(char *pStart, nmea_float_t *angleDegrees,
680  nmea_float_t *angle, int32_t *angle_fixed,
681  char *dir) {
682  char *p = pStart;
683  if (!isEmpty(p)) {
684  // get the number in DDDMM.mmmm format and break into components
685  char degreebuff[10] = {0}; // Ensure string is terminated after strncpy
686  char *e = strchr(p, '.');
687  if (e == NULL || e - p > 6)
688  return false; // no decimal point in range
689  strncpy(degreebuff, p, e - p); // get DDDMM
690  long dddmm = atol(degreebuff);
691  long degrees = (dddmm / 100); // truncate the minutes
692  long minutes = dddmm - degrees * 100; // remove the degrees
693  p = e; // start from the decimal point
694  nmea_float_t decminutes = atof(e); // the fraction after the decimal point
695  p = strchr(p, ',') + 1; // go to the next field
696 
697  // get the NSEW direction as a character
698  char nsew = 'X';
699  if (!isEmpty(p))
700  nsew = *p; // field is not empty
701  else
702  return false; // no direction provided
703 
704  // set the various numerical formats to their values
705  long fixed = degrees * 10000000 + (minutes * 10000000) / 60 +
706  (decminutes * 10000000) / 60;
707  nmea_float_t ang = degrees * 100 + minutes + decminutes;
708  nmea_float_t deg = fixed / (nmea_float_t)10000000.;
709  if (nsew == 'S' ||
710  nsew == 'W') { // fixed and deg are signed, but DDDMM.mmmm is not
711  fixed = -fixed;
712  deg = -deg;
713  }
714 
715  // reject directions that are not NSEW
716  if (nsew != 'N' && nsew != 'S' && nsew != 'E' && nsew != 'W')
717  return false;
718 
719  // reject angles that are out of range
720  if (nsew == 'N' || nsew == 'S')
721  if (abs(deg) > 90)
722  return false;
723  if (abs(deg) > 180)
724  return false;
725 
726  // store in locations passed as args
727  if (angle != NULL)
728  *angle = ang;
729  if (angle_fixed != NULL)
730  *angle_fixed = fixed;
731  if (angleDegrees != NULL)
732  *angleDegrees = deg;
733  if (dir != NULL)
734  *dir = nsew;
735  } else
736  return false; // no number
737  return true;
738 }
