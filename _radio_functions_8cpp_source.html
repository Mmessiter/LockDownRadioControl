46// **** Decompresses cc*3/4 x 16 BIT values up to cc loading only their low 12 BITS cc must be divisble by 4! ******************
47void DeComp(uint16_t* d, uint16_t* c, int cc)
48{
49 int p = 0, l = 0;
50 for (l = 0; l < (cc * 3 / 4); l += 3) {
51 d[p] = c[l] >> 4;
52 p++;
53 d[p] = (c[l] & 0xf) << 8 | c[l + 1] >> 8;
54 p++;
55 d[p] = (c[l + 1] & 0xff) << 4 | c[l + 2] >> 12;
56 p++;
57 d[p] = c[l + 2] & 0xfff;
58 p++;
59 }
60}

68void Compress(uint16_t* compressed_buf, uint16_t* uncompressed_buf, int uncompressed_size)
69{
70 int p = 0;
71 for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
72 compressed_buf[l] = uncompressed_buf[p] << 4 | uncompressed_buf[p + 1] >> 8;
73 ++p;
74 compressed_buf[l + 1] = uncompressed_buf[p] << 8 | uncompressed_buf[p + 1] >> 4;
75 ++p;
76 compressed_buf[l + 2] = uncompressed_buf[p] << 12 | uncompressed_buf[p + 1];
77 ++p;
78 ++p;
79 }
80}

122void StartMacro(uint8_t m){ // Start a macro
123 MacrosBuffer[m][MACRORUNNINGNOW] |= 1 ; // LOW BIT = "running now" flag
124 MacroStartTime[m] = millis() + ((MacrosBuffer[m][MACROSTARTTIME]) * 100); // Note its Start moment
125 MacroStopTime[m] = MacroStartTime[m] + ((MacrosBuffer[m][MACRODURATION]) * 100); // Note its Stop moment
126}
127/************************************************************************************************************/
128void RunMacro(uint8_t m){ // Move a servo to a place
129 uint32_t RightNow = millis();
130 if (RightNow >= MacroStartTime[m]) MacrosBuffer[m][MACRORUNNINGNOW] |= 2 ; // Set the ACTIVE Bit if started (BIT 1)
131 if (RightNow >= MacroStopTime[m]) MacrosBuffer[m][MACRORUNNINGNOW] &= 1 ; // Clear the ACTIVE Bit if expired (BIT 1)
132 if (MacrosBuffer[m][MACRORUNNINGNOW] & 2) SendBuffer[(MacrosBuffer[m][MACROMOVECHANNEL])-1] = map(MacrosBuffer[m][MACROMOVETOPOSITION],0,180,MINMICROS,MAXMICROS); // Do it if currently active!
133}
134/************************************************************************************************************/
135void StopMacro(uint8_t m){ // Stop a macro
136 MacrosBuffer[m][MACRORUNNINGNOW] = 0;
137}

140void ExecuteMacro(){ // Main entry point from SendData() ... START/STOP/RUN
141 uint8_t TriggerChannel = 0;
142 for (u_int8_t i = 0; i < MAXMACROS; ++i){
143// ***************************** START OR STOP ******************************
144 TriggerChannel = (MacrosBuffer[i][MACROTRIGGERCHANNEL])-1; // Down by one as channels are really 0 - 15
145 if (TriggerChannel) { // Is trigger channel non-zero?
146 if (SendBuffer[TriggerChannel] >= MAXMICROS-1){ // Is the trigger point is close to its highest value?
147 if (!MacrosBuffer[i][MACRORUNNINGNOW]) StartMacro(i); // Yes. Start if not already started
148 } else {
149 if (MacrosBuffer[i][MACRORUNNINGNOW]) StopMacro(i); // No. Stop it if it was running
150 }
151 // **************************** RUN ****************************************
152 if (MacrosBuffer[i][MACRORUNNINGNOW]) { // If running, move the servo ... if timer agrees.
153 RunMacro(i);
154 }
155 }
156 }
157}

166void SendData()
167{
168 if (NEXTION.available()) return; // was a button pressed?
169 if (millis() - TxPace <= 2) {
170 ShowComms(); // there is time to fit in these calls because there are about 5 ms spare still WHEN CONNECTED
171 ReadSwitches(); // Check switch positions
172 CheckTimer();
173 }
174 if (((millis() - TxPace) >= PACEMAKER) || (LostContactFlag)){
175 TxPace = millis();
176 GetNewChannelValues(); // Load SendBuffer with new servo positions
177 if (UseMacros) ExecuteMacro(); // Modify it if macro is running
178 if (DoSbusSendOnly) { // If buddying (SLAVE) by wire, send SBUS data down wire only and transmit nothing.
179 ReadSwitches();
180 MapToSBUS();
181 return; // no more to do here!
182 }
183 if (BuddyMaster) {GetSlaveChannelValues();} // If buddy master, check where student's sticks etc. are.
184 if (!BoundFlag && !(CurrentView == CALIBRATEVIEW) && !(CurrentView == STICKSVIEW)){
185 BufferNewPipe(); // if not yet bound, send our pipe
186 }
187 LoadPacketData(); // extra parameters appended to the data packet
188 if (LostContactFlag){
189 if ((millis() - PipeTimeout) > BINDPIPETIMEOUT) {
190 TryOtherPipe();
191 PipeTimeout = millis();
192 }
193 NextChannel = * (FHSSChPointer + random(RECONNECT_CHANNELS_COUNT) + RECONNECT_CHANNELS_START); // a **random** reconnect channel (selected from first five)
194 ShowComms(); // NEEDED WHEN NOT CONNECTED
195 ReadSwitches();
196 CheckTimer();
197 HopToNextChannel();
198 }
199 Connected = false;
200 Compress(CompressedData, SendBuffer, UNCOMPRESSEDWORDS); // Compress 32 bytes down to 24
201 Radio1.flush_rx(); // This avoids a lockup that happens when the FIFO gets full.
202 Radio1.flush_tx(); // This avoids a lockup that happens when the FIFO gets full.
203 // *************************************** SEND *************************************************************************************
204 if (Radio1.write(&CompressedData, SizeOfCompressedData)){ // **************************** ! ACTUALLY SEND DATA ! *********************************************
205 // ***************************** (Returns TRUE if ACK received) ****************************************************************************
206 ++RangeTestGoodPackets;
207 ++PacketNumber;
208 LostContactFlag = false;
209 RecentPacketsLost = 0;
210 TotalledRecentPacketsLost = 0;
211 Connected = true;
212 if (BoundFlag) GreenLedOn();
213 CheckGapsLength();
214 StartInactvityTimeout();
215 if (Radio1.isAckPayloadAvailable()){ // This 'if' is redundant
216 Radio1.read(&AckPayload, AckPayloadSize); // "sizeof" doesn't work with externs,
217 ParseAckPayload();
218 }
219 }else{
220 FailedPacket();
221 }
222 }
223}
