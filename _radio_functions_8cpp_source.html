36 // **** Decompresses cc*3/4 x 16 BIT values up to cc loading only their low 12 BITS cc must be divisble by 4! ******************
37 void DeComp(uint16_t* d, uint16_t* c, int cc)
38 {
39  int p = 0, l = 0;
40  for (l = 0; l < (cc * 3 / 4); l += 3) {
41  d[p] = c[l] >> 4;
42  p++;
43  d[p] = (c[l] & 0xf) << 8 | c[l + 1] >> 8;
44  p++;
45  d[p] = (c[l + 1] & 0xff) << 4 | c[l + 2] >> 12;
46  p++;
47  d[p] = c[l + 2] & 0xfff;
48  p++;
49  }
50 }
