22FASTRUN void Compress(uint16_t* compressed_buf, uint16_t* uncompressed_buf, uint8_t uncompressed_size)
23{
24 uint8_t p = 0;
25 for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
26 compressed_buf[l] = uncompressed_buf[p] << 4 | uncompressed_buf[p + 1] >> 8;
27 ++p;
28 compressed_buf[l + 1] = uncompressed_buf[p] << 8 | uncompressed_buf[p + 1] >> 4;
29 ++p;
30 compressed_buf[l + 2] = uncompressed_buf[p] << 12 | uncompressed_buf[p + 1];
31 ++p;
32 ++p;
33 }
34}

83void StartMacro(uint8_t m){ // Start a macro
84 MacrosBuffer[m][MACRORUNNINGNOW] |= 1 ; // LOW BIT = "running now" flag
85 MacroStartTime[m] = millis() + ((MacrosBuffer[m][MACROSTARTTIME]) * 100); // Note its Start moment
86 MacroStopTime[m] = MacroStartTime[m] + ((MacrosBuffer[m][MACRODURATION]) * 100); // Note its Stop moment
87
88}
89/************************************************************************************************************/
90void RunMacro(uint8_t m){ // Move a servo to a place
91 uint32_t RightNow = millis();
92 if (RightNow >= MacroStartTime[m]) MacrosBuffer[m][MACRORUNNINGNOW] |= 2 ; // Set the ACTIVE Bit if started (BIT 1)
93 if (RightNow >= MacroStopTime[m]) MacrosBuffer[m][MACRORUNNINGNOW] &= 1 ; // Clear the ACTIVE Bit if expired (BIT 1)
94 if (MacrosBuffer[m][MACRORUNNINGNOW] & 2) {
95 SendBuffer[(MacrosBuffer[m][MACROMOVECHANNEL])-1] = map(MacrosBuffer[m][MACROMOVETOPOSITION],0,180,MINMICROS,MAXMICROS); // Do it if currently active!
96 }
97}
98/************************************************************************************************************/
99void StopMacro(uint8_t m){ // Stop a macro
100 MacrosBuffer[m][MACRORUNNINGNOW] = 0;
101}

104void ExecuteMacro(){ // Main entry point from SendData() ... START/STOP/RUN
105 uint8_t TriggerChannel = 0;
106 for (u_int8_t i = 0; i < MAXMACROS; ++i){
107// ***************************** START OR STOP ******************************
108 TriggerChannel = (MacrosBuffer[i][MACROTRIGGERCHANNEL])-1; // Down by one as channels are really 0 - 15
109 if (TriggerChannel) { // Is trigger channel non-zero?
110 if (SendBuffer[TriggerChannel] >= MAXMICROS-1){ // Is the trigger point is close to its highest value?
111 if (!MacrosBuffer[i][MACRORUNNINGNOW]) StartMacro(i); // Yes. Start if not already started
112 } else {
113 if (MacrosBuffer[i][MACRORUNNINGNOW]) StopMacro(i); // No. Stop it if it was running
114 }
115 // **************************** RUN ****************************************
116 if (MacrosBuffer[i][MACRORUNNINGNOW]) { // If running, move the servo ... if timer agrees.
117 RunMacro(i);
118 }
119 }
120 }
121}

128FASTRUN void FailedPacket()
129{
130 int SecondsRemaining;
131 if (GapStart == 0) GapStart = millis(); // To keep track of gaps' length
132 ++RecentPacketsLost;
133 ++TotalledRecentPacketsLost; // this is to keep track of events when receiver is off
134 if (RecentPacketsLost >= LOSTCONTACTCUTOFF) {
135
136 LostContactFlag = true;
137 Reconnected = false;
138 RecentPacketsLost = 0;
139
140 if ((millis() - GapStart) > RED_LED_ON_TIME) // there's no need to blink red for every single lost packet. Only after 1/2 second of no connection.
141 {
142 if (LedWasGreen && UseLog) LogThisLongGap();
143 RedLedOn();
144 ReEnableScanButton();
145 }
146 }
147 ++LostPackets;
148 SecondsRemaining = (Inactivity_Timeout / 1000) - (millis() - Inactivity_Start) / 1000;
149 if (SecondsRemaining <= 0) digitalWrite(POWER_OFF_PIN, HIGH); // INACTIVITY POWER OFF HERE!!
150}

187FASTRUN void SendData()
188{
189 if (NEXTION.available()) return; // in case key was hit
190 if (((millis() - TxPace) >= PACEMAKER) || (LostContactFlag)){
191 TxPace = millis();
192 if (DoSbusSendOnly) {MapToSBUS();return;} // If buddying (SLAVE) by wire, send SBUS data down wire only and transmit nothing.
193 if (LostContactFlag) TryToReconnect();
194 LoadPacketData(); // extra parameters appended to the data packet
195 Connected = false; // Assume the worst until ACK is received.
196 FlushFifos();
197 if (Radio1.write(&CompressedData, SizeOfCompressedData)){ // ************************** >>>>> SEND DATA TO RX <<<<< ***************************************
198 SuccessfulPacket();

201 }
