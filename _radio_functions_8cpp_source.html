36// **** Decompresses cc*3/4 x 16 BIT values up to cc loading only their low 12 BITS cc must be divisble by 4! ******************
37void DeComp(uint16_t* d, uint16_t* c, int cc)
38{
39 int p = 0, l = 0;
40 for (l = 0; l < (cc * 3 / 4); l += 3) {
41 d[p] = c[l] >> 4;
42 p++;
43 d[p] = (c[l] & 0xf) << 8 | c[l + 1] >> 8;
44 p++;
45 d[p] = (c[l + 1] & 0xff) << 4 | c[l + 2] >> 12;
46 p++;
47 d[p] = c[l + 2] & 0xfff;
48 p++;
49 }
50}

58void Compress(uint16_t* compressed_buf, uint16_t* uncompressed_buf, int uncompressed_size)
59{
60 int p = 0;
61 for (int l = 0; l < (uncompressed_size * 3 / 4); l += 3) {
62 compressed_buf[l] = uncompressed_buf[p] << 4 | uncompressed_buf[p + 1] >> 8;
63 ++p;
64 compressed_buf[l + 1] = uncompressed_buf[p] << 8 | uncompressed_buf[p + 1] >> 4;
65 ++p;
66 compressed_buf[l + 2] = uncompressed_buf[p] << 12 | uncompressed_buf[p + 1];
67 ++p;
68 ++p;
69 }
70}

103void SendData()
104{
105 if (Nextion.available()) return; // was a button pressed?
106 if (millis() - TxPace <= 2) ShowComms(); // This call takes <2 ms, so there is time to fit in the call because there are about 5 ms spare still.
107 if ((millis() - TxPace) >= PACEMAKER) {
108 TxPace = millis();
109 get_new_channels_values(); // Load SendBuffer with new servo positions
110 if (DoSbusSendOnly) // If buddying (SLAVE) by wire, send SBUS data down wire only and transmit nothing.
111 {
112 ReadSwitches();
113 MapToSBUS();
114 return; // no more to do here!
115 }
116 if (BuddyMaster) // If buddy master, check where student's sticks etc. are.
117 {
118 GetSlaveChannelValues();
119 }
120 if (!BoundFlag && !(CurrentView == CalibrateView) && !(CurrentView == SticksView))
121 {
122 BufferNewPipe(); // if not yet bound, send our pipe
123 }
124
125
126
127 LoadPacketData(); // extra parameters appended to the data packet
128 if (LostContactFlag) {
129 if ((millis() - PipeTimeout) > BINDPIPETIMEOUT) {
130 TryOtherPipe();
131 PipeTimeout = millis();
132 }
133 NextFrequency = RECONNECT_CH;
134 HopToNextFrequency();
135 }
136 Connected = false;
137 Compress(CompressedData, SendBuffer, UNCOMPRESSEDWORDS); // Compress 32 bytes down to 24
138 Radio1.flush_rx(); // This avoids a lockup that happens when the FIFO gets full.
139 Radio1.flush_tx(); // This avoids a lockup that happens when the FIFO gets full.
140 // *************************************** SEND *************************************************************************************
141 Radio1.write(&CompressedData, SizeOfCompressedData); // ************************************** !SEND! *********************************************
142 // *************************************** SEND *************************************************************************************
143 if (Radio1.isAckPayloadAvailable())
144 {
145 Radio1.read(&AckPayload, AckPayloadSize); // "sizeof" doesn't work with externs, hence 2 new vars.
146 ++RangeTestGoodPackets;
147 ++PacketNumber;
148 LostContactFlag = false;
149 ParseAckPayload();
150 RecentPacketsLost = 0;
151 TotalledRecentPacketsLost = 0;
152 Connected = true;
153 if (BoundFlag) GreenLedOn();
154 CheckGapsLength();
155 StartInactvityTimeout();
156 }
157 else
158 {
159 FailedPacket();
160 }
161 }
162}
