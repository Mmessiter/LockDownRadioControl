364  uint32_t tStart = millis(); // as close as we can get to time char was sent
365  char c = 0;

369

371  if (gpsSwSerial) {
372  if (!gpsSwSerial->available())
373  return c;

375  }
376 #endif
377  if (gpsHwSerial) {
378  if (!gpsHwSerial->available())
379  return c;
380  c = gpsHwSerial->read();
381  }
382  if (gpsI2C) {
383  if (_buff_idx <= _buff_max) {
384  c = _i2cbuffer[_buff_idx];
385  _buff_idx++;
386  } else {
387  // refill the buffer!
388  if (gpsI2C->requestFrom((uint8_t)0x10, (uint8_t)GPS_MAX_I2C_TRANSFER,
389  (uint8_t) true) == GPS_MAX_I2C_TRANSFER) {
390  // got data!
391  _buff_max = 0;
392  char curr_char = 0;
393  for (int i = 0; i < GPS_MAX_I2C_TRANSFER; i++) {
394  curr_char = gpsI2C->read();
395  if ((curr_char == 0x0A) && (last_char != 0x0D)) {
396  // skip duplicate 0x0A's - but keep as part of a CRLF
397  continue;
398  }
399  last_char = curr_char;
400  _i2cbuffer[_buff_max] = curr_char;
401  _buff_max++;
402  }
403  _buff_max--; // back up to the last valid slot
404  if ((_buff_max == 0) && (_i2cbuffer[0] == 0x0A)) {
405  _buff_max = -1; // ahh there was nothing to read after all
406  }
407  _buff_idx = 0;
408  }
409  return c;
410  }
411  }
412 
413  if (gpsSPI) {
414  do {
415  gpsSPI->beginTransaction(gpsSPI_settings);
416  if (gpsSPI_cs >= 0) {
417  digitalWrite(gpsSPI_cs, LOW);
418  }
419  c = gpsSPI->transfer(0xFF);
420  if (gpsSPI_cs >= 0) {
421  digitalWrite(gpsSPI_cs, HIGH);
422  }
423  gpsSPI->endTransaction();
424  // skip duplicate 0x0A's - but keep as part of a CRLF
425  } while (((c == 0x0A) && (last_char != 0x0D)) ||
426  (!isprint(c) && !isspace(c)));
427  last_char = c;
428  }
429  // Serial.print(c);
430 
431  currentline[lineidx++] = c;
432  if (lineidx >= MAXLINELENGTH)
433  lineidx = MAXLINELENGTH -
434  1; // ensure there is someplace to put the next received character
435 
436  if (c == '\n') {
437  currentline[lineidx] = 0;
438 
439  if (currentline == line1) {
440  currentline = line2;
441  lastline = line1;
442  } else {
443  currentline = line1;
444  lastline = line2;
445  }

449  // Serial.println("----");

557 bool Adafruit_GPS::LOCUS_ReadStatus(void) {
558  sendCommand(PMTK_LOCUS_QUERY_STATUS);
559 
560  if (!waitForSentence("$PMTKLOG"))
561  return false;
562 
563  char *response = lastNMEA();
564  uint16_t parsed[10];
565  uint8_t i;
566 
567  for (i = 0; i < 10; i++)
568  parsed[i] = -1;
569 
570  response = strchr(response, ',');
571  for (i = 0; i < 10; i++) {
572  if (!response || (response[0] == 0) || (response[0] == '*'))
573  break;
574  response++;
575  parsed[i] = 0;
576  while ((response[0] != ',') && (response[0] != '*') && (response[0] != 0)) {
577  parsed[i] *= 10;
578  char c = response[0];
579  if (isDigit(c))
580  parsed[i] += c - '0';
581  else
582  parsed[i] = c;
583  response++;
584  }
585  }
586  LOCUS_serial = parsed[0];
587  LOCUS_type = parsed[1];
588  if (isAlpha(parsed[2])) {
589  parsed[2] = parsed[2] - 'a' + 10;
590  }
591  LOCUS_mode = parsed[2];
592  LOCUS_config = parsed[3];
593  LOCUS_interval = parsed[4];
594  LOCUS_distance = parsed[5];
595  LOCUS_speed = parsed[6];
596  LOCUS_status = !parsed[7];
597  LOCUS_records = parsed[8];
