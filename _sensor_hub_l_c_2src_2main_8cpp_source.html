87 float RdataOut = 42;
88 union { float Val32;uint8_t Val8[4]; } Rdata;
89 switch (ParameterNumber) {
90 case 0:

105 break;
106 case 4:

108 strcpy (RdataID,SPD);
109 break;
110 case 5:
111 RdataOut = GPSCourse; // Course
112 strcpy (RdataID,COR);
113 break;
114 case 6:
115 RdataOut = GPSCourseTo; // Course to
116 strcpy (RdataID,CTO);
117 break;
118 case 7:
119 RdataOut = GPSDistanceTo; // Distance to
120 strcpy (RdataID,DTO);
121 break;
122 case 8:
123 RdataOut = (float) GPSHours; // Hours
124 strcpy (RdataID,HRS);
125 break;
126 case 9:
127 RdataOut = (float) GPSMins; // Mins
128 strcpy (RdataID,MNS);
129 break;
130 case 10:
131 RdataOut = (float) GPSSecs; // Secs
132 strcpy (RdataID,SEC);
133 break;
134 case 11:
135 RdataOut = (float) GPSSatellites; // Sats
136 strcpy (RdataID,SAT);

139 RdataOut = BaroAltitude; // ALT FROM BMP280
140 strcpy (RdataID,BLT);

143 RdataOut = BaroTemperature; // TEMPERATURE FROM BMP280
144 strcpy (RdataID,TMP);

157 break;
158 case 17:
159 RdataOut = GPSYear; // YEAR
160 strcpy (RdataID,YER);
161 break;
162 default:
163 break;
164 }
165 Rdata.Val32 = RdataOut;
166 for (i = 0; i < IDLEN; ++i) {
167 Wire1.write(RdataID[i]);
168 }
169 for (i = 0; i < 4; ++i) {
170 Wire1.write(Rdata.Val8[i]);
171 }
172 ParameterNumber++;
173 if (ParameterNumber > MAXPARAMS) ParameterNumber = 0;

177void DoTheRequest(){
178char MRK[] = "MRK";
179char MAY[] = "MAY";
180char QNH[] = "QNH";
181char RCV[10] = "???????";
182
183union {uint16_t Val16; uint8_t Val8[2];} Uqnh;
184
185ReceiveRequestFlag = false;
186
187 for (uint8_t i = 0; i < 6; ++i) {
188 if (Wire1.available()) {
189 RCV[i]= Wire1.read(); // Get one byte at a time
190 }
191 }
192
193 RCV[3] = 0; // Add a terminator
194 Serial.print (millis());
195 Serial.print (" ");
196 Serial.println (RCV);
197
198 if (strcmp(LastRequest, RCV) == 0) return; // temp bug fix!! (lots of phantom repeat calls?????)
199 strcpy(LastRequest,RCV);
200
201 if (strcmp(MRK,RCV) == 0) { // Match first 3 chars with MRK?
202 DestinationLat = GPSLatitude; // Mark current location
203 DestinationLng = GPSLongitude; // Mark current location
204 return;
205 }
206
207 if (strcmp(MAY,RCV) == 0) { // Match first 3 chars with MAY?
208 DestinationLat = MAYSLANE_LAT; // Mark MAYS LANE location
209 DestinationLng = MAYSLANE_LON; // Mark MAYS LANE location
210 return;
211 }
212
213 if (strcmp(QNH,RCV) == 0) { // Match first 3 chars with QNH?
214 Uqnh.Val8[0] = RCV[4]; // 16 bit value for "Qnh" sent in bytes 4 and 5
215 Uqnh.Val8[1] = RCV[5];
216 Qnh = Uqnh.Val16;
217 return;
218 }
219}
